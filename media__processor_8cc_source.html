62 void HandleGenericFFmpegError(int code) {
63  // See //third_party/ffmpeg/src/libavutil/error.h
64  switch (code) {
65  case AVERROR_BSF_NOT_FOUND:
66  case AVERROR_DECODER_NOT_FOUND:
67  case AVERROR_DEMUXER_NOT_FOUND:
68  case AVERROR_ENCODER_NOT_FOUND:
69  case AVERROR_FILTER_NOT_FOUND:
70  case AVERROR_MUXER_NOT_FOUND:
71  case AVERROR_OPTION_NOT_FOUND:
72  case AVERROR_STREAM_NOT_FOUND:
73  // This should be handled by VideoController::AddSource.
74  LOG(DFATAL) << "Unable to find media handler: " << ErrStr(code);
75  break;
76  case AVERROR_BUFFER_TOO_SMALL:
77  case AVERROR_EOF:
78  case AVERROR_INVALIDDATA:
79  case AVERROR_INPUT_CHANGED:
80  case AVERROR_OUTPUT_CHANGED:
81  case AVERROR(EAGAIN):
82  case AVERROR(EINVAL):
83  case AVERROR(ENOMEM):
84  // Calling code should handle these codes.
85  LOG(DFATAL) << "Special error not handled: " << ErrStr(code);
86  break;
87  case AVERROR_HTTP_BAD_REQUEST:
88  case AVERROR_HTTP_UNAUTHORIZED:
89  case AVERROR_HTTP_FORBIDDEN:
90  case AVERROR_HTTP_NOT_FOUND:
91  case AVERROR_HTTP_OTHER_4XX:
92  case AVERROR_HTTP_SERVER_ERROR:
93  case AVERROR_PROTOCOL_NOT_FOUND:
94  // We don't use FFmpeg's networking, so this shouldn't happen.
95  LOG(DFATAL) << "Unexpected networking error: " << ErrStr(code);
96  break;
97  case AVERROR_BUG:
98  case AVERROR_BUG2:
99  case AVERROR_PATCHWELCOME:
100  LOG(DFATAL) << "Bug inside FFmpeg: " << ErrStr(code);
101  break;
102 
103  case AVERROR_EXIT:
104  case AVERROR_EXTERNAL:
105  case AVERROR_UNKNOWN:
106  LOG(ERROR) << "Unknown error inside FFmpeg: " << ErrStr(code);
107  break;
108 
109  default:
110  LOG(DFATAL) << "Unknown error: " << ErrStr(code);
111  break;
112  }

115 const AVCodec* FindCodec(AVCodecID codec_id) {
116 #ifdef ENABLE_HARDWARE_DECODE
117  const AVCodec* hybrid = nullptr;
118  const AVCodec* external = nullptr;
119  void* opaque = nullptr;
120  while (const AVCodec* codec = av_codec_iterate(&opaque)) {
121  if (codec->id == codec_id && av_codec_is_decoder(codec)) {
122  if (codec->capabilities & AV_CODEC_CAP_HARDWARE)
123  return codec;
124 
125  if (codec->capabilities & AV_CODEC_CAP_HYBRID) {
126  // Keep the hybrid as a fallback, but try to find a hardware-only one.
127  hybrid = codec;
128  } else if (codec->wrapper_name) {
129  // This is an external codec, which may be provided by the OS. Fallback
130  // to this if nothing else.
131  external = codec;
132  }
133  }
134  }
135  if (hybrid)
136  return hybrid;
137  if (external)
138  return external;
139 #endif
140  return avcodec_find_decoder(codec_id);
141 }

148 std::vector<uint8_t> CreatePssh(AVEncryptionInitInfo* info) {
149  // 4 box size
150  // 4 box type
151  // 1 version
152  // 3 flags
153  // 16 system_id
154  // if (version > 0)
155  // 4 key_id_count
156  // for (key_id_count)
157  // 16 key_id
158  // 4 data_size
159  // [data_size] data
160  DCHECK_EQ(info->system_id_size, 16u);
161  size_t pssh_size = info->data_size + 32;
162  if (info->num_key_ids) {
163  DCHECK_EQ(info->key_id_size, 16u);
164  pssh_size += 4 + info->num_key_ids * 16;
165  }
166 
167 #define WRITE_INT(ptr, num) *reinterpret_cast<uint32_t*>(ptr) = htonl(num)
168 
169  std::vector<uint8_t> pssh(pssh_size, 0);
170  uint8_t* ptr = pssh.data();
171  WRITE_INT(ptr, pssh_size);
172  WRITE_INT(ptr + 4, 0x70737368); // 'pssh'
173  WRITE_INT(ptr + 8, info->num_key_ids ? 0x01000000 : 0);
174  std::memcpy(ptr + 12, info->system_id, 16);
175  ptr += 28;
176  if (info->num_key_ids) {
177  WRITE_INT(ptr, info->num_key_ids);
178  ptr += 4;
179  for (uint32_t i = 0; i < info->num_key_ids; i++) {
180  std::memcpy(ptr, info->key_ids[i], 16);
181  ptr += 16;
182  }
183  }
184 
185  WRITE_INT(ptr, info->data_size);
186  std::memcpy(ptr + 4, info->data, info->data_size);
187  DCHECK_EQ(ptr + info->data_size + 4, pssh.data() + pssh_size);
188 
189 #undef WRITE_INT
190  return pssh;
191 }

256  Status ReinitDemuxer(std::unique_lock<Mutex>* lock) {
257  avformat_close_input(&demuxer_ctx_);
258  avio_flush(io_);
259 
260  AVFormatContext* new_ctx;
261  // Create the new demuxer without the lock held because the method will
262  // block while waiting for the init segment.
263  lock->unlock();
264  const Status status = CreateDemuxer(&new_ctx);
265  if (status != Status::Success)
266  return status;
267  lock->lock();
268  demuxer_ctx_ = new_ctx;
269 
270  AVStream* stream = demuxer_ctx_->streams[0];
271  auto* params = avcodec_parameters_alloc();
272  if (!params) {
273  return Status::OutOfMemory;
274  }
275  const int copy_code = avcodec_parameters_copy(params, stream->codecpar);
276  if (copy_code < 0) {
277  avcodec_parameters_free(&params);
278  HandleGenericFFmpegError(copy_code);
279  return Status::CannotOpenDemuxer;
280  }
281  codec_params_.push_back(params);
282  time_scales_.push_back(stream->time_base);
283 
284  return Status::Success;
285  }

287  Status CreateDemuxer(AVFormatContext** context) {
288  // Allocate a demuxer context and set it to use the IO context.
289  AVFormatContext* demuxer = avformat_alloc_context();
290  if (!demuxer) {
291  return Status::OutOfMemory;
292  }
293  demuxer->pb = io_;
294  // If we enable the probes, in encrypted content we'll get logs about being
295  // unable to parse the content; however, if we disable the probes, we won't
296  // get accurate frame durations, which can cause problems.
297  // TODO: Find a way to conditionally disable parsing or to suppress the
298  // logs for encrypted content (since the errors there aren't fatal).
299  // demuxer->probesize = 0;
300  // demuxer->max_analyze_duration = 0;
301 
302  // To enable extremely verbose logging:
303  // demuxer->debug = 1;
304 
305  // Parse encryption info for WebM; ignored for other demuxers.
306  AVDictionary* dict = nullptr;
307  CHECK_EQ(av_dict_set_int(&dict, "parse_encryption", 1, 0), 0);
308 
309  AVInputFormat* format = av_find_input_format(container_.c_str());
310  CHECK(format) << "Should have checked for support before creating.";
311  const int open_code = avformat_open_input(&demuxer, nullptr, format, &dict);
312  av_dict_free(&dict);
313  if (open_code < 0) {
314  // If avformat_open_input fails, it will free and reset |demuxer|.
315  DCHECK(!demuxer);
316 
317  if (open_code == AVERROR(ENOMEM))
318  return Status::OutOfMemory;
319  if (open_code == AVERROR_INVALIDDATA)
320  return Status::InvalidContainerData;
321 
322  HandleGenericFFmpegError(open_code);
323  return Status::CannotOpenDemuxer;
324  }
325 
326  const int find_code = avformat_find_stream_info(demuxer, nullptr);
327  if (find_code < 0) {
328  avformat_close_input(&demuxer);
329  if (find_code == AVERROR(ENOMEM))
330  return Status::OutOfMemory;
331  if (find_code == AVERROR_INVALIDDATA)
332  return Status::InvalidContainerData;
333 
334  HandleGenericFFmpegError(find_code);
335  return Status::CannotOpenDemuxer;
336  }
337 
338  if (demuxer->nb_streams == 0) {
339  avformat_close_input(&demuxer);
340  return Status::NoStreamsFound;
341  }
342  if (demuxer->nb_streams > 1) {
343  avformat_close_input(&demuxer);
344  return Status::MultiplexedContentFound;
345  }
346 
347  *context = demuxer;
348  return Status::Success;
349  }

351  Status InitializeDemuxer(std::function<size_t(uint8_t*, size_t)> on_read,
352  std::function<void()> on_reset_read) {
353  std::unique_lock<Mutex> lock(mutex_);
354 
355  on_read_ = std::move(on_read);
356  on_reset_read_ = std::move(on_reset_read);
357 
358  // Allocate a context for custom IO.
359  // NOTE: The buffer may be reallocated/resized by libavformat later.
360  // It is always our responsibility to free it later with av_free.
361  io_ = avio_alloc_context(
362  reinterpret_cast<unsigned char*>(av_malloc(kInitialBufferSize)),
363  kInitialBufferSize,
364  0, // write_flag (read-only)
365  this, // opaque user data
366  &Impl::ReadCallback,
367  nullptr, // write callback (read-only)
368  nullptr); // seek callback (linear reads only)
369  if (!io_) {
370  return Status::OutOfMemory;
371  }
372 
373  received_frame_ = av_frame_alloc();
374  if (!received_frame_) {
375  return Status::OutOfMemory;
376  }
377 
378  const Status reinit_status = ReinitDemuxer(&lock);
379  if (reinit_status == Status::Success)
380  signal_.SignalAll();
381 
382  return reinit_status;
383  }

385  Status ReadDemuxedFrame(std::unique_ptr<BaseFrame>* frame) {
386  AVPacket pkt;
387  int ret = av_read_frame(demuxer_ctx_, &pkt);
388  if (ret == AVERROR_SHAKA_RESET_DEMUXER) {
389  // Special case for Shaka where we need to reinit the demuxer.
390  VLOG(1) << "Reinitializing demuxer";
391  on_reset_read_();
392 
393  std::unique_lock<Mutex> lock(mutex_);
394  const Status reinit_status = ReinitDemuxer(&lock);
395  if (reinit_status != Status::Success)
396  return reinit_status;
397  ret = av_read_frame(demuxer_ctx_, &pkt);
398  }
399  if (ret < 0) {
400  av_packet_unref(&pkt);
401  if (ret == AVERROR_EOF)
402  return Status::EndOfStream;
403  if (ret == AVERROR(ENOMEM))
404  return Status::OutOfMemory;
405  if (ret == AVERROR_INVALIDDATA)
406  return Status::InvalidContainerData;
407 
408  HandleGenericFFmpegError(ret);
409  return Status::UnknownError;
410  }
411 
412  UpdateEncryptionInitInfo();
413 
414  // Ignore discard flags. The demuxer will set this when we try to read
415  // content behind media we have already read.
416  pkt.flags &= ~AV_PKT_FLAG_DISCARD;
417 
418  VLOG(3) << "Read frame at dts=" << pkt.dts;
419  DCHECK_EQ(pkt.stream_index, 0);
420  DCHECK_EQ(demuxer_ctx_->nb_streams, 1u);
421  frame->reset(FFmpegEncodedFrame::MakeFrame(&pkt, demuxer_ctx_->streams[0],
422  codec_params_.size() - 1,
423  timestamp_offset_));
424  // No need to unref |pkt| since it was moved into the encoded frame.
425  return *frame ? Status::Success : Status::OutOfMemory;
426  }

428  Status InitializeDecoder(size_t stream_id, bool allow_hardware) {
429  const AVCodecParameters* params = codec_params_[stream_id];
430  const char* codec_name = avcodec_get_name(params->codec_id);
431  if (codec_ != codec_name) {
432  LOG(ERROR) << "Mismatch between codec string and media. Codec string: '"
433  << codec_ << "', media codec: '" << codec_name << "' (0x"
434  << std::hex << params->codec_id << ")";
435  return Status::DecoderMismatch;
436  }
437 
438  const AVCodec* decoder = allow_hardware
439  ? FindCodec(params->codec_id)
440  : avcodec_find_decoder(params->codec_id);
441  CHECK(decoder) << "Should have checked support already";
442 #ifdef ENABLE_HARDWARE_DECODE
443  AVHWDeviceType hw_type = AV_HWDEVICE_TYPE_NONE;
444  hw_pix_fmt_ = AV_PIX_FMT_NONE;
445  if (allow_hardware) {
446  for (int i = 0;; i++) {
447  const AVCodecHWConfig* config = avcodec_get_hw_config(decoder, i);
448  if (!config) {
449 # ifdef FORCE_HARDWARE_DECODE
450  if (!decoder->wrapper_name) {
451  LOG(DFATAL) << "No hardware-accelerators available for codec: "
452  << codec_;
453  return Status::DecoderFailedInit;
454  }
455 # endif
456  LOG(INFO) << "No hardware-accelerators available, using decoder: "
457  << decoder->name;
458  break;
459  }
460 
461  if (config->methods & AV_CODEC_HW_CONFIG_METHOD_HW_DEVICE_CTX) {
462  LOG(INFO) << "Using decoder: " << decoder->name
463  << ", with hardware accelerator: "
464  << av_hwdevice_get_type_name(config->device_type);
465  hw_type = config->device_type;
466  hw_pix_fmt_ = config->pix_fmt;
467  break;
468  }
469  }
470  }
471 #endif
472 
473  avcodec_free_context(&decoder_ctx_);
474  decoder_ctx_ = avcodec_alloc_context3(decoder);
475  if (!decoder_ctx_) {
476  return Status::OutOfMemory;
477  }
478 
479  const int param_code = avcodec_parameters_to_context(decoder_ctx_, params);
480  if (param_code < 0) {
481  if (param_code == AVERROR(ENOMEM)) {
482  return Status::OutOfMemory;
483  }
484 
485  HandleGenericFFmpegError(param_code);
486  return Status::DecoderFailedInit;
487  }
488  decoder_ctx_->thread_count = 0; // Default is 1; 0 means auto-detect.
489  decoder_ctx_->opaque = this;
490  decoder_ctx_->pkt_timebase = time_scales_[stream_id];
491 
492 #ifdef ENABLE_HARDWARE_DECODE
493  // If using a hardware accelerator, initialize it now.
494  av_buffer_unref(&hw_device_ctx_);
495  if (allow_hardware && hw_type != AV_HWDEVICE_TYPE_NONE) {
496  const int hw_device_code =
497  av_hwdevice_ctx_create(&hw_device_ctx_, hw_type, nullptr, nullptr, 0);
498  if (hw_device_code < 0) {
499  if (hw_device_code == AVERROR(ENOMEM)) {
500  return Status::OutOfMemory;
501  }
502 
503  HandleGenericFFmpegError(hw_device_code);
504  return Status::DecoderFailedInit;
505  }
506  decoder_ctx_->get_format = &GetPixelFormat;
507  decoder_ctx_->hw_device_ctx = av_buffer_ref(hw_device_ctx_);
508  }
509 #endif
510 
511  const int open_code = avcodec_open2(decoder_ctx_, decoder, nullptr);
512  if (open_code < 0) {
513  if (open_code == AVERROR(ENOMEM))
514  return Status::OutOfMemory;
515 #if defined(ENABLE_HARDWARE_DECODE) && !defined(FORCE_HARDWARE_DECODE)
516  if (allow_hardware) {
517  LOG(WARNING) << "Failed to initialize hardware decoder, falling back "
518  "to software.";
519  return InitializeDecoder(stream_id, false);
520  }
521 #endif
522 
523  HandleGenericFFmpegError(open_code);
524  return Status::DecoderFailedInit;
525  }
526 
527  decoder_stream_id_ = stream_id;
528  return Status::Success;
529  }

531  Status ReadFromDecoder(size_t stream_id, const FFmpegEncodedFrame* frame,
532  std::vector<std::unique_ptr<BaseFrame>>* decoded) {
533  while (true) {
534  const int code = avcodec_receive_frame(decoder_ctx_, received_frame_);
535  if (code == AVERROR(EAGAIN) || code == AVERROR_EOF)
536  return Status::Success;
537  if (code == AVERROR_INVALIDDATA)
538  return Status::InvalidCodecData;
539  if (code < 0) {
540  HandleGenericFFmpegError(code);
541  return Status::UnknownError;
542  }
543 
544  const double timescale = av_q2d(time_scales_[stream_id]);
545  const int64_t timestamp = received_frame_->best_effort_timestamp;
546  const double offset =
547  frame ? frame->timestamp_offset() : prev_timestamp_offset_;
548  const double time = frame && timestamp == AV_NOPTS_VALUE
549  ? frame->pts
550  : timestamp * timescale + offset;
551  auto* new_frame = FFmpegDecodedFrame::CreateFrame(
552  received_frame_, time, frame ? frame->duration : 0);
553  if (!new_frame) {
554  return Status::OutOfMemory;
555  }
556  decoded->emplace_back(new_frame);
557  }
558  }

560  Status DecodeFrame(double /* cur_time */, const BaseFrame* base_frame,
561  eme::Implementation* cdm,
562  std::vector<std::unique_ptr<BaseFrame>>* decoded) {
563  decoded->clear();
564  DCHECK(!base_frame ||
565  base_frame->frame_type() == FrameType::FFmpegEncodedFrame);
566  auto* frame = static_cast<const FFmpegEncodedFrame*>(base_frame);
567 
568  if (!frame && !decoder_ctx_) {
569  // If there isn't a decoder, there is nothing to flush.
570  return Status::Success;
571  }
572 
573  if (frame) {
574  // Wait for the signal before acquiring the lock to avoid a deadlock.
575  signal_.GetValue();
576  std::unique_lock<Mutex> lock(mutex_);
577 
578  if (!decoder_ctx_ || frame->stream_id() != decoder_stream_id_) {
579  VLOG(1) << "Reconfiguring decoder";
580  // Flush the old decoder to get any existing frames.
581  if (decoder_ctx_) {
582  const int send_code = avcodec_send_packet(decoder_ctx_, nullptr);
583  if (send_code != 0) {
584  if (send_code == AVERROR(ENOMEM))
585  return Status::OutOfMemory;
586  if (send_code == AVERROR_INVALIDDATA)
587  return Status::InvalidCodecData;
588 
589  HandleGenericFFmpegError(send_code);
590  return Status::UnknownError;
591  }
592  const Status read_result =
593  ReadFromDecoder(decoder_stream_id_, nullptr, decoded);
594  if (read_result != Status::Success)
595  return read_result;
596  }
597 
598  const Status init_result = InitializeDecoder(frame->stream_id(), true);
599  if (init_result != Status::Success)
600  return init_result;
601  }
602 
603  prev_timestamp_offset_ = frame->timestamp_offset();
604  }
605 
606 
607  // If the encoded frame is encrypted, decrypt it first.
608  AVPacket decrypted_packet{};
609  util::Finally free_decrypted_packet(
610  std::bind(&av_packet_unref, &decrypted_packet));
611  const AVPacket* frame_to_send = frame ? frame->raw_packet() : nullptr;
612  if (frame && frame->is_encrypted()) {
613  if (!cdm) {
614  LOG(WARNING) << "No CDM given for encrypted frame";
615  return Status::KeyNotFound;
616  }
617 
618  int code = av_new_packet(&decrypted_packet, frame->raw_packet()->size);
619  if (code == 0)
620  code = av_packet_copy_props(&decrypted_packet, frame->raw_packet());
621  if (code == AVERROR(ENOMEM))
622  return Status::OutOfMemory;
623  if (code < 0) {
624  HandleGenericFFmpegError(code);
625  return Status::UnknownError;
626  }
627 
628  Status decrypt_status = frame->Decrypt(cdm, &decrypted_packet);
629  if (decrypt_status != Status::Success)
630  return decrypt_status;
631  frame_to_send = &decrypted_packet;
632  }
633 
634  bool sent_frame = false;
635  while (!sent_frame) {
636  // If we get EAGAIN, we should read some frames and try to send again.
637  const int send_code = avcodec_send_packet(decoder_ctx_, frame_to_send);
638  if (send_code == 0) {
639  sent_frame = true;
640  } else if (send_code == AVERROR_EOF) {
641  // If we get EOF, this is either a flush or we are closing. Either way,
642  // stop. If this is a flush, we can't reuse the decoder, so reset it.
643  ResetDecoder();
644  break;
645  } else if (send_code != AVERROR(EAGAIN)) {
646  if (send_code == AVERROR(ENOMEM))
647  return Status::OutOfMemory;
648  if (send_code == AVERROR_INVALIDDATA)
649  return Status::InvalidCodecData;
650 
651  HandleGenericFFmpegError(send_code);
