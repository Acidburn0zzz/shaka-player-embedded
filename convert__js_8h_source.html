71 template <typename Number>
72 struct ConvertHelper<Number, _number_identifier> {
73  static bool FromJsValue(Handle<JsValue> source, Number* dest) {
74  // Number types.
75  const JSValueType type = GetValueType(source);
76  if (type != JSValueType::Number && type != JSValueType::NumberObject) {
77  return false;
78  }
79  const double value = NumberFromValue(source);
80 
81  if (std::numeric_limits<Number>::has_infinity &&
82  std::abs(value) == std::numeric_limits<Number>::infinity()) {
83  // OK.
84  } else if (value < std::numeric_limits<Number>::lowest() ||
85  value > std::numeric_limits<Number>::max()) {
86  return false;
87  }
88 
89  // JavaScript numbers will be intentionally truncated if stored as native
90  // integers.
91  *dest = static_cast<Number>(value);
92  return true;
93  }
94 
95  static ReturnVal<JsValue> ToJsValue(Number source) {
96 #if defined(USING_V8)
97  return v8::Number::New(GetIsolate(), source);
98 #elif defined(USING_JSC)
99  return JSValueMakeNumber(GetContext(), source);
100 #endif
101  }
102 };

115 template <typename T>
116 struct ConvertHelper<shaka::optional<T>, void> {
117  static bool FromJsValue(Handle<JsValue> source, shaka::optional<T>* dest) {
118  if (IsNullOrUndefined(source)) {
119  dest->reset();
120  return true;

139 template <typename... Types>
140 struct ConvertHelper<shaka::variant<Types...>> {
141  private:
142  template <size_t I, typename = void>
143  struct Helper {
144  using T = variant_alternative_t<I, shaka::variant<Types...>>;
145 
146  static bool FromJsValue(Handle<JsValue> source,
147  shaka::variant<Types...>* dest) {
148  T temp;
149  if (ConvertHelper<T>::FromJsValue(source, &temp)) {
150  dest->template emplace<I>(std::move(temp));
151  return true;
152  }
153  return Helper<I + 1>::FromJsValue(source, dest);
154  }
155 
156  static ReturnVal<JsValue> ToJsValue(
157  const shaka::variant<Types...>& source) {
158  if (source.index() == I)
159  return ConvertHelper<T>::ToJsValue(get<I>(source));
160  else
161  return Helper<I + 1>::ToJsValue(source);

197 template <typename T>
198 struct ConvertHelper<std::vector<T>, void> {
199  static bool FromJsValue(Handle<JsValue> source, std::vector<T>* dest) {
200  if (GetValueType(source) != JSValueType::Array)
201  return false;
202 
203  LocalVar<JsObject> array(UnsafeJsCast<JsObject>(source));
204  const size_t length = ArrayLength(array);
205  // Store the data in a temp array since this should not modify |*dest| on
206  // failure.
207  std::vector<T> temp(length);
208  DCHECK_EQ(length, temp.size());
209  for (size_t i = 0; i < length; i++) {
210  LocalVar<JsValue> item(GetArrayIndexRaw(array, i));
211  if (!ConvertHelper<T>::FromJsValue(item, &temp[i]))
212  return false;
213  }
214 
215  swap(temp, *dest);
216  return true;
217  }
218 
219  static ReturnVal<JsValue> ToJsValue(const std::vector<T>& source) {
220  LocalVar<JsObject> ret(CreateArray(source.size()));
221  for (size_t i = 0; i < source.size(); i++) {
222  LocalVar<JsValue> item(ConvertHelper<T>::ToJsValue(source[i]));
223  SetArrayIndexRaw(ret, i, item);
224  }
225  return RawToJsValue(ret);
226  }
227 };

243 template <typename T>
244 struct ConvertHelper<T*, void> {
245  static_assert(std::is_base_of<BackingObject, T>::value,
246  "Don't accept raw pointers as arguments, use RefPtr<T>.");
247 
248  static ReturnVal<JsValue> ToJsValue(T* source) {
249  // We cannot implicitly convert a T* to a RefPtr<T> because the compiler
250  // cannot deduce the type parameter. This allows us to pass raw pointers
251  // (e.g. |this|).
252  if (!source)
253  return JsNull();
254  else
255  return source->JsThis();
