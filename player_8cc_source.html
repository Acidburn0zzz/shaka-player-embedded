90 Error ConvertError(Handle<JsValue> except) {
91  if (!IsObject(except))
92  return Error(ErrorType::NonShakaError, ConvertToString(except));
93 
94  LocalVar<JsObject> obj = UnsafeJsCast<JsObject>(except);
95  LocalVar<JsValue> message_member = GetMemberRaw(obj, "message");
96  // Rather than checking for the type of the exception, assume that if it has
97  // a 'name' field, then it is an exception.
98  LocalVar<JsValue> name_member = GetMemberRaw(obj, "name");
99  if (!IsNullOrUndefined(name_member)) {
100  const std::string message =
101  ConvertToString(name_member) + ": " + ConvertToString(message_member);
102  return Error(ErrorType::NonShakaError, message);
103  }
104 
105  LocalVar<JsValue> code = GetMemberRaw(obj, "code");
106  LocalVar<JsValue> category = GetMemberRaw(obj, "category");
107  if (GetValueType(code) != JSValueType::Number ||
108  GetValueType(category) != JSValueType::Number) {
109  return Error(ErrorType::NonShakaError, ConvertToString(except));
110  }
111 
112  LocalVar<JsValue> severity_val = GetMemberRaw(obj, "severity");
113  int severity = 0;
114  if (GetValueType(severity_val) == JSValueType::Number)
115  severity = static_cast<int>(NumberFromValue(severity_val));
116 
117  std::string message;
118  if (IsNullOrUndefined(message_member)) {
119  message = "Shaka Error, Category: " + ConvertToString(category) +
120  ", Code: " + ConvertToString(code);
121  } else {
122  message = ConvertToString(message_member);
123  }
124  return Error(static_cast<int>(NumberFromValue(category)),
125  static_cast<int>(NumberFromValue(code)), severity, message);
126 }

128 Converter<void>::variant_type CallMemberFunction(Handle<JsObject> that,
129  const std::string& name,
130  int argc,
131  LocalVar<JsValue>* argv,
132  LocalVar<JsValue>* result) {
133  LocalVar<JsValue> member = GetMemberRaw(that, name);
134  if (GetValueType(member) != JSValueType::Function) {
135  return Error(ErrorType::BadMember,
136  "The member '" + name + "' is not a function.");
137  }
138 
139  LocalVar<JsValue> result_or_except;
140  LocalVar<JsFunction> member_func = UnsafeJsCast<JsFunction>(member);
141  if (!InvokeMethod(member_func, that, argc, argv, &result_or_except)) {
142  return ConvertError(result_or_except);
143  }
144 
145  if (result)
146  *result = result_or_except;
147  return monostate();
148 }

150 Converter<void>::variant_type AttachEventListener(
151  Handle<JsObject> player, const std::string& name, Player::Client* client,
152  std::function<void(Handle<JsObject> event)> handler) {
153  const std::function<void(optional<Any>)> callback = [=](optional<Any> event) {
154  // We can't accept or use events::Event since Shaka player raises fake
155  // events. So manually look for the properties.
156  LocalVar<JsValue> event_val = ToJsValue(event);
157  if (!IsObject(event_val)) {
158  client->OnError(
159  Error(ErrorType::NonShakaError, ConvertToString(event_val)));
160  return;
161  }
162 
163  LocalVar<JsObject> event_obj = UnsafeJsCast<JsObject>(event_val);
164  handler(event_obj);
165  };
166  LocalVar<JsFunction> callback_js = CreateStaticFunction("", "", callback);
167  LocalVar<JsValue> arguments[] = {ToJsValue(name), RawToJsValue(callback_js)};
168  return CallMemberFunction(player, "addEventListener", 2, arguments, nullptr);
169 }

194  Converter<void>::future_type Initialize(js::mse::HTMLVideoElement* video,
195  Client* client) {
196  // This function can be called immediately after the JsManager
197  // constructor. Since the Environment might not be setup yet, run this in
198  // an internal task so we know it is ready.
199  const auto callback = [=]() -> Converter<void>::variant_type {
200  LocalVar<JsValue> player_ctor = GetDescendant(
201  JsEngine::Instance()->global_handle(), {"shaka", "Player"});
202  if (GetValueType(player_ctor) != JSValueType::Function) {
203  LOG(ERROR) << "Cannot get 'shaka.Player' object; is "
204  "shaka-player.compiled.js corrupted?";
205  return Error(ErrorType::BadMember,
206  "The constructor 'shaka.Player' is not found.");
207  }
208  LocalVar<JsFunction> player_ctor_func =
209  UnsafeJsCast<JsFunction>(player_ctor);
210 
211  LocalVar<JsValue> result_or_except;
212  LocalVar<JsValue> args[] = {video->JsThis()};
213  if (!InvokeConstructor(player_ctor_func, 1, args, &result_or_except)) {
214  return ConvertError(result_or_except);
215  }
216 
217  player_ = UnsafeJsCast<JsObject>(result_or_except);
218  return AttachListeners(player_, client);
219  };
220  return JsManagerImpl::Instance()
221  ->MainThread()
222  ->AddInternalTask(TaskPriority::Internal, "Player ctor",
223  PlainCallbackTask(callback))
224  ->future();
225  }

228  template <typename Ret, typename... Args>
229  typename Converter<Ret>::future_type CallPlayerMethod(
230  const std::string& name, const Args&... args) const {
231  const auto callback = [=]() -> typename Converter<Ret>::variant_type {
232  LocalVar<JsValue> result;
233  LocalVar<JsValue> js_args[] = {ToJsValue(args)..., JsUndefined()};
234  auto error =
235  CallMemberFunction(player_, name, sizeof...(args), js_args, &result);
236  if (holds_alternative<Error>(error))
237  return get<Error>(error);
238 
239  return Converter<Ret>::Convert(name, result);
240  };
241  return JsManagerImpl::Instance()
242  ->MainThread()
243  ->AddInternalTask(TaskPriority::Internal, "Player." + name,
244  PlainCallbackTask(callback))
245  ->future();
246  }

253  template <typename Ret, typename... Args>
254  typename Converter<Ret>::future_type CallPlayerPromiseMethod(
255  const std::string& name, Args&&... args) {
256  // TODO: Combine with CallPlayerMethod. These are separate since this uses
257  // a std::promise for the return value rather than the internal task future.
258  auto promise =
259  std::make_shared<std::promise<typename Converter<Ret>::variant_type>>();
260  const auto callback = [this, promise, name, args...]() {
261  LocalVar<JsValue> result;
262  LocalVar<JsValue> js_args[] = {ToJsValue(args)..., JsUndefined()};
263  auto error =
264  CallMemberFunction(player_, name, sizeof...(args), js_args, &result);
265  if (holds_alternative<Error>(error)) {
266  promise->set_value(get<Error>(error));
267  return;
268  }
269 
270  auto js_promise = Converter<Promise>::Convert(name, result);
271  if (holds_alternative<Error>(js_promise)) {
272  promise->set_value(get<Error>(js_promise));
273  return;
274  }
275 
276  get<Promise>(js_promise)
277  .Then(
278  [promise, name](Any res) {
279  LocalVar<JsValue> value = res.ToJsValue();
280  promise->set_value(Converter<Ret>::Convert(name, value));
281  },
282  [promise](Any except) {
283  LocalVar<JsValue> val = except.ToJsValue();
284  promise->set_value(ConvertError(val));
285  });
286  };
287  JsManagerImpl::Instance()
288  ->MainThread()
289  ->AddInternalTask(TaskPriority::Internal, "Player." + name,
290  PlainCallbackTask(callback))
291  ->future();
292  return promise->get_future().share();
293  }

295  template <typename T>
296  typename Converter<T>::future_type GetConfigValue(
297  const std::string& name_path) {
298  const auto callback = [=]() -> typename Converter<T>::variant_type {
299  LocalVar<JsValue> configuration;
300  auto error = CallMemberFunction(player_, "getConfiguration", 0, nullptr,
301  &configuration);
302  if (holds_alternative<Error>(error))
303  return get<Error>(error);
304 
305  // Split the name path on periods.
306  std::vector<std::string> components = util::StringSplit(name_path, '.');
307 
308  // Navigate through the path.
309  auto result =
310  GetDescendant(UnsafeJsCast<JsObject>(configuration), components);
311 
312  return Converter<T>::Convert(name_path, result);
313  };
314  return JsManagerImpl::Instance()
315  ->MainThread()
316  ->AddInternalTask(TaskPriority::Internal, "Player.getConfiguration",
317  PlainCallbackTask(callback))
318  ->future();
319  }

321  private:
322  Converter<void>::variant_type AttachListeners(Handle<JsObject> player,
323  Client* client) {
324 #define ATTACH(name, call) \
325  do { \
326  const auto ret = AttachEventListener(player, name, client, call); \
327  if (holds_alternative<Error>(ret)) \
328  return get<Error>(ret); \
329  } while (false)
330 
331  const auto on_error = [=](Handle<JsObject> event) {
332  LocalVar<JsValue> detail = GetMemberRaw(event, "detail");
333  client->OnError(ConvertError(detail));
334  };
335  ATTACH("error", on_error);
336 
337  const auto on_buffering = [=](Handle<JsObject> event) {
338  LocalVar<JsValue> is_buffering = GetMemberRaw(event, "buffering");
339  bool is_buffering_bool;
340  if (FromJsValue(is_buffering, &is_buffering_bool)) {
341  client->OnBuffering(is_buffering_bool);
342  } else {
343  client->OnError(Error(ErrorType::NonShakaError,
344  "Bad 'buffering' event from JavaScript Player"));
345  }
346  };
347  ATTACH("buffering", on_buffering);
348 
349 #undef ATTACH
350  return {};
351  }
