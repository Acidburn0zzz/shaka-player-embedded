50 SDL_AudioFormat SDLFormatFromFFmpeg(AVSampleFormat format) {
51  // Try to use the same format to avoid work by swresample.
52  switch (format) {
53  case AV_SAMPLE_FMT_U8:
54  case AV_SAMPLE_FMT_U8P:
55  return AUDIO_U8;
56  case AV_SAMPLE_FMT_S16:
57  case AV_SAMPLE_FMT_S16P:
58  return AUDIO_S16SYS;
59  case AV_SAMPLE_FMT_S32:
60  case AV_SAMPLE_FMT_S32P:
61  return AUDIO_S32SYS;
62  case AV_SAMPLE_FMT_FLT:
63  case AV_SAMPLE_FMT_FLTP:
64  return AUDIO_F32SYS;
65 
66  case AV_SAMPLE_FMT_DBL:
67  case AV_SAMPLE_FMT_DBLP: {
68  LOG_ONCE(WARNING) << "SDL doesn't support double-precision audio "
69  << "formats, converting to floats.";
70  return AUDIO_F32SYS;
71  }
72  case AV_SAMPLE_FMT_S64:
73  case AV_SAMPLE_FMT_S64P: {
74  LOG_ONCE(WARNING) << "SDL doesn't support 64-bit audio "
75  << "formats, converting to 32-bit.";
76  return AUDIO_S32SYS;
77  }
78 
79  default:
80  LOG(DFATAL) << "Unknown audio sample format: " << format;
81  return AUDIO_S32SYS;
82  }
83 }

85 AVSampleFormat FFmpegFormatFromSDL(int format) {
86  // Note that AUDIO_*SYS is an alias for either AUDIO_*LSB or AUDIO_*MSB.
87  switch (format) {
88  case AUDIO_U8:
89  return AV_SAMPLE_FMT_U8;
90  case AUDIO_S16LSB:
91  case AUDIO_S16MSB:
92  if (format != AUDIO_S16SYS) {
93  LOG(ERROR) << "swresample doesn't support non-native endian audio";
94  return AV_SAMPLE_FMT_NONE;
95  }
96  return AV_SAMPLE_FMT_S16;
97  case AUDIO_S32LSB:
98  case AUDIO_S32MSB:
99  if (format != AUDIO_S32SYS) {
100  LOG(ERROR) << "swresample doesn't support non-native endian audio";
101  return AV_SAMPLE_FMT_NONE;
102  }
103  return AV_SAMPLE_FMT_S32;
104  case AUDIO_F32LSB:
105  case AUDIO_F32MSB:
106  if (format != AUDIO_F32SYS) {
107  LOG(ERROR) << "swresample doesn't support non-native endian audio";
108  return AV_SAMPLE_FMT_NONE;
109  }
110  return AV_SAMPLE_FMT_FLT;
111 
112  case AUDIO_S8:
113  LOG(ERROR) << "swresample doesn't support signed 8-bit audio.";
114  return AV_SAMPLE_FMT_NONE;
115  case AUDIO_U16LSB:
116  case AUDIO_U16MSB:
117  LOG(ERROR) << "swresample doesn't support unsigned 16-bit audio";
118  return AV_SAMPLE_FMT_NONE;
119 
120  default:
121  LOG(DFATAL) << "Unknown audio sample format: " << format;
122  return AV_SAMPLE_FMT_S32;
123  }

126 int64_t GetChannelLayout(int num_channels) {
127  // See |channels| in https://wiki.libsdl.org/SDL_AudioSpec.
128  switch (num_channels) {
129  case 1:
130  return AV_CH_LAYOUT_MONO;
131  case 2:
132  return AV_CH_LAYOUT_STEREO;
133  case 4:
134  return AV_CH_LAYOUT_QUAD;
135  case 6:
136  return AV_CH_LAYOUT_5POINT1;
137 
138  default:
139  LOG(DFATAL) << "Unsupported channel count: " << num_channels;
140  return AV_CH_LAYOUT_STEREO;

201 void AudioRenderer::ThreadMain() {
202  std::unique_lock<Mutex> lock(mutex_);
203  while (!shutdown_) {
204  if (need_reset_) {
205  if (audio_device_ != 0) {
206  SDL_CloseAudioDevice(audio_device_);
207  audio_device_ = 0;
208  }
209 
210  cur_time_ = get_time_();
211  auto base_frame = stream_->GetDecodedFrames()->GetFrameAfter(cur_time_);
212  if (!base_frame) {
213  util::Unlocker<Mutex> unlock(&lock);
214  util::Clock::Instance.SleepSeconds(0.01);
215  continue;
216  }
217 
218  CHECK(base_frame->frame_type() == FrameType::FFmpegDecodedFrame);
219  auto* frame = static_cast<const FFmpegDecodedFrame*>(base_frame.get());
220 
221  if (!InitDevice(frame))
222  return;
223 
224  SDL_PauseAudioDevice(audio_device_, 0);
225  need_reset_ = false;
226  }
227 
228  on_reset_.ResetAndWaitWhileUnlocked(lock);
229  }
230 }

232 bool AudioRenderer::InitDevice(const FFmpegDecodedFrame* frame) {
233  if (!SDL_WasInit(SDL_INIT_AUDIO)) {
234  SDL_SetMainReady();
235  if (SDL_InitSubSystem(SDL_INIT_AUDIO) < 0) {
236  LOG(ERROR) << "Error initializing SDL: " << SDL_GetError();
237  return false;
238  }
239  }
240 
241  memset(&audio_spec_, 0, sizeof(audio_spec_));
242  audio_spec_.freq = frame->raw_frame()->sample_rate;
243  audio_spec_.format = SDLFormatFromFFmpeg(frame->sample_format());
244  audio_spec_.channels = static_cast<Uint8>(frame->raw_frame()->channels);
245  audio_spec_.samples = static_cast<Uint16>(frame->raw_frame()->nb_samples *
246  frame->raw_frame()->channels);
247  audio_spec_.callback = &OnAudioCallback;
248  audio_spec_.userdata = this;
249  audio_device_ =
250  SDL_OpenAudioDevice(nullptr, 0, &audio_spec_, &obtained_audio_spec_,
251  SDL_AUDIO_ALLOW_ANY_CHANGE);
252  if (audio_device_ == 0) {
253  LOG(ERROR) << "Error opening audio device: " << SDL_GetError();
254  return false;
255  }
256 
257  // SDL may change the format so we get hardware acceleration. Make sure
258  // to use the format SDL expects.
259  const AVSampleFormat av_sample_format =
260  FFmpegFormatFromSDL(obtained_audio_spec_.format);
261  if (av_sample_format == AV_SAMPLE_FMT_NONE)
262  return false;
263 
264  swr_ctx_ = swr_alloc_set_opts(
265  swr_ctx_,
266  GetChannelLayout(obtained_audio_spec_.channels), // out_ch_layout
267  av_sample_format, // out_sample_fmt
268  obtained_audio_spec_.freq, // out_sample_rate
269  frame->raw_frame()->channel_layout, // in_ch_layout
270  frame->sample_format(), // in_sample_fmt
271  frame->raw_frame()->sample_rate, // in_sample_rate
272  0, // log_offset,
273  nullptr); // log_ctx
274  if (!swr_ctx_) {
275  LOG(ERROR) << "Unable to allocate swrescale context.";
276  return false;
277  }
278 
279  // Minimum difference before changing samples to match timestamps.
280  av_opt_set_double(swr_ctx_, "min_comp", 0.01, 0);
281  // Maximum factor to adjust existing samples by.
282  av_opt_set_double(swr_ctx_, "max_soft_comp", 0.01, 0);
283  // Minimum difference before applying hard compensation (adding/dropping
284  // samples).
285  av_opt_set_double(swr_ctx_, "min_hard_comp", 0.1, 0);
286  // Sync samples to timestamps.
287  av_opt_set_double(swr_ctx_, "async", 1, 0);
288  // Change volume to this value.
289  av_opt_set_double(swr_ctx_, "rematrix_volume", volume_, 0);
290 
291  swr_init(swr_ctx_);
292  return true;
293 }

300 void AudioRenderer::AudioCallback(uint8_t* data, int size) {
301  std::unique_lock<Mutex> lock(mutex_);
302 
303  if (cur_time_ >= 0)
304  stream_->GetDecodedFrames()->Remove(0, cur_time_ - 0.2);
305 
306  const double playback_rate = get_playback_rate_();
307  // TODO: Support playback rate by using atemp filter.
308  DCHECK(playback_rate == 0 || playback_rate == 1)
309  << "Only playbackRate of 0 and 1 are supported.";
310  if (need_reset_ || is_seeking_ || volume_ == 0 || playback_rate <= 0 ||
311  playback_rate > kMaxPlaybackRate) {
312  memset(data, obtained_audio_spec_.silence, size);
313  return;
314  }
315 
316  const AVSampleFormat av_sample_format =
317  FFmpegFormatFromSDL(obtained_audio_spec_.format);
318  const int sample_size =
319  av_get_bytes_per_sample(av_sample_format) * obtained_audio_spec_.channels;
320  int size_in_samples = size / sample_size;
321  DCHECK_EQ(size % sample_size, 0);
322 
323  const double now_time = get_time_();
324  if (cur_time_ >= 0) {
325  // |cur_time_ - delay| represents the playhead time that is about to be
326  // played.
327  const double delay = swr_get_delay(swr_ctx_, 1000) / 1000.0;
328  if (cur_time_ - delay < now_time - kMaxDelay) {
329  // The next frame being played is from too long ago; so simulate a seek to
330  // play the audio at the playhead.
331  cur_time_ = -1;
332  }
333  }
334 
335  if (cur_time_ < 0) {
336  cur_time_ = now_time;
337  // swr will adjust samples to match their expected timestamps; reset the
338  // context on seek so it doesn't break with the new timestamps.
339  swr_init(swr_ctx_);
340  }
341 
342  // Flush existing data before reading more frames.
343  const int initial_sample_count =
344  swr_convert(swr_ctx_, &data, size_in_samples, nullptr, 0);
345  if (initial_sample_count < 0) {
346  memset(data, 0, size);
347  return;
348  }
349  DCHECK_LE(initial_sample_count, size_in_samples);
350  size_in_samples -= initial_sample_count;
351  data += initial_sample_count * sample_size;
352 
353  while (size_in_samples > 0) {
354  auto base_frame = stream_->GetDecodedFrames()->GetFrameAfter(cur_time_);
355  if (!base_frame)
356  break;
357 
358  CHECK(base_frame->frame_type() == FrameType::FFmpegDecodedFrame);
359  auto* frame = static_cast<const FFmpegDecodedFrame*>(base_frame.get());
360 
361  // If the source changed, we need to reset. If the new frame has a lower
362  // sample rate or channel count, we can just use swresample to change
363  // these. If they are higher, we want to try to create a new device so we
364  // get the benefits.
365  if (frame->raw_frame()->sample_rate > audio_spec_.freq ||
366  frame->raw_frame()->channels > audio_spec_.channels ||
367  SDLFormatFromFFmpeg(frame->sample_format()) != audio_spec_.format) {
368  need_reset_ = true;
369  on_reset_.SignalAll();
370  break;
371  }
372  if (frame->raw_frame()->sample_rate != audio_spec_.freq) {
373  av_opt_set_int(swr_ctx_, "in_sample_rate",
374  frame->raw_frame()->sample_rate, 0);
375  swr_init(swr_ctx_);
376  audio_spec_.freq = frame->raw_frame()->sample_rate;
377  }
378  if (frame->raw_frame()->channels != audio_spec_.channels) {
379  av_opt_set_int(swr_ctx_, "in_channel_layout",
380  GetChannelLayout(frame->raw_frame()->channels), 0);
381  swr_init(swr_ctx_);
382  audio_spec_.channels = static_cast<Uint8>(frame->raw_frame()->channels);
383  }
384 
385  // Assume the first byte in the array will be played "right-now", or at
386  // |now_time|. This is technically not correct, but the delay shouldn't be
387  // noticeable.
388  const auto pts =
389  static_cast<uint64_t>(frame->pts * obtained_audio_spec_.freq *
390  frame->raw_frame()->sample_rate);
391  // Swr will adjust the audio so the next sample will happen at |pts|.
392  if (swr_next_pts(swr_ctx_, pts) < 0)
393  break;
394 
395  const int samples_read =
396  swr_convert(swr_ctx_, &data, size_in_samples,
397  const_cast<const uint8_t**>(frame->data()), // NOLINT
398  frame->raw_frame()->nb_samples);
399  if (samples_read < 0)
400  break;
401 
402  DCHECK_LE(samples_read, size_in_samples);
403  size_in_samples -= samples_read;
404  data += samples_read * sample_size;
405 
406  cur_time_ = frame->pts;
407  }
408 
409  // Set any remaining data to silence in the event of errors.
410  memset(data, obtained_audio_spec_.silence, size_in_samples * sample_size);
411 }
