39 void Video::Initialize() {
40  // This can be called immediately after the JsManager constructor. Since the
41  // Environment might not be setup yet, run this in an internal task so we know
42  // it is ready.
43  const auto callback = [this]() {
44  impl_->inner =
45  new js::mse::HTMLVideoElement(js::dom::Document::GetGlobalDocument());
46  };
47  JsManagerImpl::Instance()
48  ->MainThread()
49  ->AddInternalTask(TaskPriority::Internal, "Video init",
50  PlainCallbackTask(callback))
51  ->GetValue();
52 }

87 std::vector<TextTrack> Video::TextTracks() {
88  std::vector<Member<js::mse::TextTrack>> original =
89  impl_->GetMemberVariable(&JSVideo::text_tracks);
90 
91  // For each element in [begin, end), transform it using |callback| and
92  // insert the result at the back of |text_tracks|.
93  std::vector<TextTrack> text_tracks;
94  auto begin = original.begin();
95  auto end = original.end();
96  const auto callback = [](const Member<js::mse::TextTrack> track) {
97  return TextTrack(track);
98  };
99  std::transform(begin, end, std::back_inserter(text_tracks), callback);
100  return text_tracks;
101 }
