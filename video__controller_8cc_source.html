35 std::string FormatSize(const FrameBuffer* buffer) {
36  const char* kSuffixes[] = {"", " KB", " MB", " GB", " TB"};
37  size_t size = buffer->EstimateSize();
38  for (const char* suffix : kSuffixes) {
39  if (size < 3 * 1024)
40  return std::to_string(size) + suffix;
41  size /= 1024;
42  }
43  LOG(FATAL) << "Size too large to print.";
44 }

46 std::string FormatBuffered(const FrameBuffer* buffer) {
47  std::string ret;

49  if (!ret.empty())
50  ret += ", ";
51  ret += util::StringPrintf("%.2f - %.2f", range.start, range.end);
52  }
53  return "[" + ret + "]";

140 Frame VideoController::DrawFrame(double* delay) {
141  std::unique_lock<SharedMutex> lock(mutex_);
142  Source* source = GetSource(SourceType::Video);
143  if (!source || !source->renderer)
144  return Frame();
145 
146  int dropped_frame_count = 0;
147  bool is_new_frame = false;
148  Frame ret =
149  source->renderer->DrawFrame(&dropped_frame_count, &is_new_frame, delay);
150  quality_info_.droppedVideoFrames += dropped_frame_count;
151  quality_info_.totalVideoFrames += dropped_frame_count;
152  if (is_new_frame)
153  quality_info_.totalVideoFrames++;
154 
155 #ifndef NDEBUG
156  static uint64_t last_print_time = 0;
157  static uint64_t last_dropped_frame_count = 0;
158  const uint64_t cur_time = util::Clock::Instance.GetMonotonicTime();
159  if (quality_info_.droppedVideoFrames != last_dropped_frame_count &&
160  cur_time - last_print_time > 250) {
161  const int delta_frames =
162  quality_info_.droppedVideoFrames - last_dropped_frame_count;
163  LOG(INFO) << "Dropped " << delta_frames << " frames.";
164  last_print_time = cur_time;
165  last_dropped_frame_count = quality_info_.droppedVideoFrames;
166  }
167 #endif
168 
169  return ret;
170 }

181 Status VideoController::AddSource(const std::string& mime_type,
182  SourceType* source_type) {
183  std::unique_lock<SharedMutex> lock(mutex_);
184  using namespace std::placeholders; // NOLINT
185  std::string container;
186  std::string codec;
187  if (!ParseMimeAndCheckSupported(mime_type, source_type, &container, &codec))
188  return Status::NotSupported;
189  if (sources_.count(*source_type) != 0)
190  return Status::NotAllowed;
191 
192  std::unique_ptr<Source> source(new Source(
193  *source_type, &pipeline_, container, codec,
194  MainThreadCallback(on_waiting_for_key_),
195  std::bind(&VideoController::OnEncryptedInitData, this, _1, _2, _3),
196  std::bind(&PipelineManager::GetCurrentTime, &pipeline_),
197  std::bind(&VideoController::GetPlaybackRate, this),
198  std::bind(&VideoController::OnError, this, *source_type, _1),
199  std::bind(&VideoController::OnLoadMeta, this, *source_type)));
200  if (source->renderer) {
201  if (*source_type == SourceType::Audio)
202  static_cast<AudioRenderer*>(source->renderer.get())->SetVolume(volume_);
203  }
204  source->decoder.SetCdm(cdm_);
205  sources_.emplace(*source_type, std::move(source));
206  return Status::Success;
207 }

237 void VideoController::EndOfStream() {
238  double duration = 0;
239  {
240  util::shared_lock<SharedMutex> lock(mutex_);
241  for (auto& pair : sources_) {
242  auto buffered = pair.second->stream.GetBufferedRanges();
243  // Use the maximum duration of any stream as the total media duration.
244  // See: https://w3c.github.io/media-source/#end-of-stream-algorithm
245  if (!buffered.empty())
246  duration = std::max(duration, buffered.back().end);
247  }
248  }
249 
250  pipeline_.SetDuration(duration);
251 }

316 void VideoController::OnLoadMeta(SourceType type) {
317  bool done = true;
318  double duration;
319  {
320  util::shared_lock<SharedMutex> lock(mutex_);
321  DCHECK_EQ(1u, sources_.count(type));
322  DCHECK(!sources_.at(type)->ready);
323  sources_.at(type)->ready = true;
324  duration = sources_.at(type)->processor.duration();
325 
326  for (auto& pair : sources_) {
327  if (!pair.second->ready)
328  done = false;
329  }
330  }
331  if (std::isnan(pipeline_.GetDuration()))
332  pipeline_.SetDuration(duration == 0 ? HUGE_VAL : duration);
333  if (done)
334  pipeline_.DoneInitializing();
335 }
