77 void DecoderThread::ThreadMain() {
78  while (!shutdown_.load(std::memory_order_acquire)) {
79  const double cur_time = get_time_();
80  double last_time = last_frame_time_.load(std::memory_order_acquire);
81 
82  LockedFrameList::Guard frame;
83  if (std::isnan(last_time)) {
84  processor_->ResetDecoder();
85  frame = stream_->GetDemuxedFrames()->GetKeyFrameBefore(cur_time);
86  } else {
87  frame = stream_->GetDemuxedFrames()->GetFrameAfter(last_time);
88  }
89 
90  if (stream_->DecodedAheadOf(cur_time) > kDecodeBufferSize) {
91  std::this_thread::sleep_for(std::chrono::milliseconds(25));
92  continue;
93  }
94  if (!frame) {
95  if (!std::isnan(last_time) &&
96  last_time + kEndDelta >= pipeline_->GetDuration() &&
97  !did_flush_.load(std::memory_order_acquire)) {
98  // If this is the last frame, pass the null to DecodeFrame, which will
99  // flush the decoder.
100  did_flush_.store(true, std::memory_order_release);
101  } else {
102  std::this_thread::sleep_for(std::chrono::milliseconds(25));
103  continue;
104  }
105  }
106 
107  std::vector<std::unique_ptr<BaseFrame>> decoded;
108  eme::Implementation* cdm = cdm_.load(std::memory_order_acquire);
109  const Status decode_status =
110  processor_->DecodeFrame(cur_time, frame.get(), cdm, &decoded);
111  if (decode_status == Status::KeyNotFound) {
112  // If we don't have the required key, signal the <video> and wait.
113  if (!raised_waiting_event_) {
114  raised_waiting_event_ = true;
115  on_waiting_for_key_();
116  }
117  std::this_thread::sleep_for(std::chrono::milliseconds(100));
118  continue;
119  }
120  if (decode_status != Status::Success) {
121  on_error_(decode_status);
122  break;
123  }
124 
125  raised_waiting_event_ = false;
126  const double last_pts = decoded.empty() ? -1 : decoded.back()->pts;
127  for (auto& decoded_frame : decoded)
128  stream_->GetDecodedFrames()->AppendFrame(std::move(decoded_frame));
129 
130  if (frame) {
131  // Don't change the |last_frame_time_| if it was reset to NAN while this
132  // was running.
133  const bool updated = last_frame_time_.compare_exchange_strong(
134  last_time, frame->dts, std::memory_order_acq_rel);
135  if (updated && last_pts >= cur_time) {
136  bool expected = true;
137  if (is_seeking_.compare_exchange_strong(expected, false,
138  std::memory_order_acq_rel)) {
139  seek_done_();
140  }
141  }
142  }
143  }
144 }
