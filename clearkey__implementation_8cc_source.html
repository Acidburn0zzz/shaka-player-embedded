32 bool ParseKeyIds(const Data& data, std::vector<std::string>* base64_keyids) {
33  LocalVar<JsValue> data_val =
34  ParseJsonString(std::string(data.data(), data.data() + data.size()));
35  if (!IsObject(data_val)) {
36  LOG(ERROR) << "Init data is not valid JSON.";
37  return false;
38  }
39  LocalVar<JsObject> data_obj = UnsafeJsCast<JsObject>(data_val);
40  LocalVar<JsValue> kids = GetMemberRaw(data_obj, "kids");
41  if (GetValueType(kids) != JSValueType::Array) {
42  LOG(ERROR) << "Init data doesn't have valid 'kids' member.";
43  return false;
44  }
45 
46  LocalVar<JsObject> kids_obj = UnsafeJsCast<JsObject>(kids);
47  const size_t kid_count = ArrayLength(kids_obj);
48  base64_keyids->reserve(kid_count);
49  for (size_t i = 0; i < kid_count; i++) {
50  LocalVar<JsValue> entry = GetArrayIndexRaw(kids_obj, i);
51  if (GetValueType(entry) != JSValueType::String) {
52  LOG(ERROR) << "Init data doesn't have valid 'kids' member.";
53  return false;
54  }
55  base64_keyids->emplace_back(ConvertToString(entry));
56  }
57  return true;
58 }

60 bool ParsePssh(const Data& data, std::vector<std::string>* base64_keyids) {
61  // 4 box size
62  // 4 box type
63  // 4 version + flags
64  // 16 system-id
65  // if (version > 0)
66  // 4 key id count
67  // for (key id count)
68  // 16 key id
69  // 4 data size
70  // [data size] data
71  constexpr const size_t kSystemIdSize = 16;
72  constexpr const size_t kKeyIdSize = 16;
73  constexpr const uint8_t kCommonSystemId[] = {
74  0x10, 0x77, 0xef, 0xec, 0xc0, 0xb2, 0x4d, 0x02,
75  0xac, 0xe3, 0x3c, 0x1e, 0x52, 0xe2, 0xfb, 0x4b};
76 
77  util::BufferReader reader(data.data(), data.size());
78  while (!reader.empty()) {
79  const size_t box_start_remaining = reader.BytesRemaining();
80  const uint32_t box_size = reader.ReadUint32();
81 
82  if (reader.ReadUint32() != 0x70737368) { // 'pssh'
83  LOG(ERROR) << "Init data is not a PSSH box";
84  return false;
85  }
86 
87  const uint8_t version = reader.ReadUint8();
88  reader.Skip(3);
89 
90  uint8_t system_id[kSystemIdSize];
91  if (reader.Read(system_id, kSystemIdSize) != kSystemIdSize) {
92  LOG(ERROR) << "Truncated init data";
93  return false;
94  }
95 
96  if (memcmp(system_id, kCommonSystemId, kSystemIdSize) != 0) {
97  VLOG(1) << "Ignoring non-common PSSH box";
98  const size_t bytes_read = box_start_remaining - reader.BytesRemaining();
99  reader.Skip(box_size - bytes_read);
100  continue;
101  }
102 
103  if (version == 0) {
104  LOG(ERROR) << "PSSH version 0 is not supported for clear-key";
105  return false;
106  }
107 
108  const uint32_t key_id_count = reader.ReadUint32();
109  if (reader.BytesRemaining() / kKeyIdSize < key_id_count) {
110  LOG(ERROR) << "Truncated init data";
111  return false;
112  }
113 
114  base64_keyids->reserve(key_id_count);
115  for (uint32_t i = 0; i < key_id_count; i++) {
116  uint8_t key_id[kKeyIdSize];
117  if (reader.Read(key_id, kKeyIdSize) != kKeyIdSize) {
118  LOG(ERROR) << "Truncated init data";
119  return false;
120  }
121  base64_keyids->emplace_back(
122  js::Base64::EncodeUrl(ByteString(key_id, key_id + kKeyIdSize)));
123  }
124  return true;
125  }
126 
127  LOG(ERROR) << "No PSSH box with common system ID found";
128  return false;
129 }

131 bool ParseAndGenerateRequest(MediaKeyInitDataType type, const Data& data,
132  std::string* message) {
133  std::vector<std::string> key_ids;
134  switch (type) {
135  case MediaKeyInitDataType::KeyIds:
136  if (!ParseKeyIds(data, &key_ids))
137  return false;
138  break;
139  case MediaKeyInitDataType::Cenc:
140  if (!ParsePssh(data, &key_ids))
141  return false;
142  break;
143 
144  default:
145  LOG(ERROR) << "Init data type not supported.";
146  return false;
147  }
148 
149  std::string ids_json;
150  for (const std::string& id : key_ids) {
151  if (ids_json.empty())
152  ids_json = '"' + id + '"';
153  else
154  ids_json += R"(,")" + id + '"';
155  }
156  *message = R"({"kids":[)" + ids_json + R"(],"type":"temporary"})";
157  return true;
158 }

160 // This needs to be a template to access the private type |Session::Key|.
161 template <typename KeyType>
162 bool ParseResponse(const Data& data, std::list<KeyType>* keys) {
163  LocalVar<JsValue> data_val =
164  ParseJsonString(std::string(data.data(), data.data() + data.size()));
165  if (!IsObject(data_val)) {
166  LOG(ERROR) << "License response is not valid JSON.";
167  return false;
168  }
169  LocalVar<JsObject> data_obj = UnsafeJsCast<JsObject>(data_val);
170  LocalVar<JsValue> keys_val = GetMemberRaw(data_obj, "keys");
171  if (GetValueType(keys_val) != JSValueType::Array) {
172  LOG(ERROR) << "License response doesn't contain a valid 'keys' member.";
173  return false;
174  }
175 
176  LocalVar<JsObject> keys_array = UnsafeJsCast<JsObject>(keys_val);
177  const size_t key_count = ArrayLength(keys_array);
178  for (size_t i = 0; i < key_count; i++) {
179  LocalVar<JsValue> entry = GetArrayIndexRaw(keys_array, i);
180  if (!IsObject(entry)) {
181  LOG(ERROR) << "License response doesn't contain a valid 'keys' member.";
182  return false;
183  }
184  LocalVar<JsObject> entry_obj = UnsafeJsCast<JsObject>(entry);
185 
186  LocalVar<JsValue> k_val = GetMemberRaw(entry_obj, "k");
187  LocalVar<JsValue> kid_val = GetMemberRaw(entry_obj, "kid");
188  if (GetValueType(k_val) != JSValueType::String ||
189  GetValueType(kid_val) != JSValueType::String) {
190  LOG(ERROR) << "License response contains invalid key object.";
191  return false;
192  }
193 
194  ExceptionOr<ByteString> k = js::Base64::DecodeUrl(ConvertToString(k_val));
195  ExceptionOr<ByteString> kid =
196  js::Base64::DecodeUrl(ConvertToString(kid_val));
197  if (holds_alternative<js::JsError>(k) ||
198  holds_alternative<js::JsError>(kid)) {
199  LOG(ERROR) << "License response contains invalid base-64 encoding.";
200  return false;
201  }
202  if (get<ByteString>(k).size() != 16 || get<ByteString>(kid).size() != 16) {
203  LOG(ERROR) << "Key or key ID is not correct size.";
204  return false;
205  }
206  keys->emplace_back(std::move(get<ByteString>(kid)),
207  std::move(get<ByteString>(k)));
208  }
209 
210  return true;
211 }

246 void ClearKeyImplementation::CreateSessionAndGenerateRequest(
247  EmePromise promise, std::function<void(const std::string&)> set_session_id,
248  MediaKeySessionType session_type, MediaKeyInitDataType init_data_type,
249  Data data) {
250  DCHECK(session_type == MediaKeySessionType::Temporary);
251 
252  std::unique_lock<std::mutex> lock(mutex_);
253 
254  const std::string session_id = std::to_string(++cur_session_id_);
255  // The indexer will create a new object since it doesn't exist.
256  Session* session = &sessions_[session_id];
257  DCHECK(!session->callable);
258 
259  std::string message;
260  if (!ParseAndGenerateRequest(init_data_type, data, &message)) {
261  promise.Reject(ExceptionType::TypeError,
262  "Invalid initialization data given.");
263  return;
264  }
265 
266  session->callable = true;
267  set_session_id(session_id);
268  helper_->OnMessage(session_id, MediaKeyMessageType::LicenseRequest,
269  reinterpret_cast<const uint8_t*>(message.c_str()),
270  message.size());
271  promise.Resolve();
272 }

280 void ClearKeyImplementation::Update(const std::string& session_id,
281  EmePromise promise, Data data) {
282  std::unique_lock<std::mutex> lock(mutex_);
283 
284  if (sessions_.count(session_id) == 0) {
285  promise.Reject(ExceptionType::InvalidState,
286  "Unable to find given session ID.");
287  return;
288  }
289  Session* session = &sessions_.at(session_id);
290 
291  if (!session->callable) {
292  promise.Reject(ExceptionType::InvalidState, "Not expecting an update.");
293  return;
294  }
295 
296  std::list<Session::Key> keys;
297  if (!ParseResponse(data, &keys)) {
298  promise.Reject(ExceptionType::InvalidState, "Invalid response data.");
299  return;
300  }
301 
302  session->callable = false;
303  // Move all keys into the session.
304  session->keys.splice(session->keys.end(), std::move(keys));
305  helper_->OnKeyStatusChange(session_id);
306  promise.Resolve();
307 }

322 DecryptStatus ClearKeyImplementation::Decrypt(
323  EncryptionScheme scheme, EncryptionPattern pattern, uint32_t block_offset,
324  const uint8_t* key_id, size_t key_id_size, const uint8_t* iv,
325  size_t iv_size, const uint8_t* data, size_t data_size,
326  uint8_t* dest) const {
327  std::unique_lock<std::mutex> lock(mutex_);
328 
329  const std::vector<uint8_t> key_id_vec(key_id, key_id + key_id_size);
330  const std::vector<uint8_t>* key = nullptr;
331  for (auto& session_pair : sessions_) {
332  for (auto& cur_key : session_pair.second.keys) {
333  if (cur_key.key_id == key_id_vec) {
334  key = &cur_key.key;
335  break;
336  }
337  }
338  }
339  if (!key) {
340  LOG(ERROR) << "Unable to find key ID: "
341  << util::ToHexString(key_id, key_id_size);
342  return DecryptStatus::KeyNotFound;
343  }
344 
345  util::Decryptor decryptor(scheme, *key,
346  std::vector<uint8_t>{iv, iv + iv_size});
347 
348  size_t num_bytes_read = 0;
349  block_offset %= AES_BLOCK_SIZE;
350  if (block_offset != 0) {
351  if (pattern.clear_blocks != 0) {
352  LOG(ERROR) << "Cannot have block offset when using pattern encryption";
353  return DecryptStatus::OtherError;
354  }
355 
356  num_bytes_read = std::min<int>(data_size, AES_BLOCK_SIZE - block_offset);
357  if (!decryptor.DecryptPartialBlock(data, num_bytes_read, block_offset,
358  dest)) {
359  return DecryptStatus::OtherError;
360  }
361  }
362 
363  if (pattern.clear_blocks != 0) {
364  const size_t protected_size = AES_BLOCK_SIZE * pattern.encrypted_blocks;
365  const size_t clear_size = AES_BLOCK_SIZE * pattern.clear_blocks;
366  const size_t pattern_size_in_blocks =
367  pattern.encrypted_blocks + pattern.clear_blocks;
368  const size_t data_size_in_blocks =
369  (data_size - num_bytes_read) / AES_BLOCK_SIZE;
370  for (size_t i = 0; i < data_size_in_blocks / pattern_size_in_blocks; i++) {
371  if (!decryptor.Decrypt(data + num_bytes_read, protected_size,
372  dest + num_bytes_read)) {
373  return DecryptStatus::OtherError;
374  }
375  num_bytes_read += protected_size;
376 
377  memcpy(dest + num_bytes_read, data + num_bytes_read, clear_size);
378  num_bytes_read += clear_size;
379  }
380 
381  // If the last pattern block isn't big enough for the whole
382  // encrypted_blocks, then it is ignored.
383  if (data_size_in_blocks % pattern_size_in_blocks >=
384  pattern.encrypted_blocks) {
385  if (!decryptor.Decrypt(data + num_bytes_read, protected_size,
386  dest + num_bytes_read)) {
387  return DecryptStatus::OtherError;
388  }
389  num_bytes_read += protected_size;
390  }
391 
392  memcpy(dest + num_bytes_read, data + num_bytes_read,
393  data_size - num_bytes_read);
394  } else {
395  if (!decryptor.Decrypt(data + num_bytes_read, data_size - num_bytes_read,
396  dest + num_bytes_read)) {
397  return DecryptStatus::OtherError;
398  }
399  }
400 
401  return DecryptStatus::Success;
402 }
