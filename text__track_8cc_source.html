56 void TextTrack::CheckForCueChange(double newTime, double oldTime) {
57  std::unique_lock<Mutex> lock(mutex_);
58  if (mode_ == TextTrackMode::Disabled)
59  return;
60 
61  const double earlier = std::min(oldTime, newTime);
62  const double later = std::max(oldTime, newTime);
63  for (const auto& cue : cues) {
64  if ((cue->startTime() > earlier && cue->startTime() <= later) ||
65  (cue->endTime() > earlier && cue->endTime() <= later)) {
66  // A cue just began, or ended.
67  ScheduleEvent<events::Event>(EventType::CueChange);
68  return;
69  }
70  }
71 }
