139 void Environment::Install() {
140  RegisterDefaultKeySystems();
141 
142  impl_.reset(new Impl);
143 
144  media::MediaProcessor::Initialize();
145 
146  JsEngine* engine = JsEngine::Instance();
147  LocalVar<JsValue> window(engine->global_value());
148  SetMemberRaw(engine->global_handle(), "window", window);
149 
150  // Shaka Player registers an "error" handler on "window".
151  // TODO: Create a global object type so we can have window events.
152  RegisterGlobalFunction("addEventListener", &DummyMethod);
153  RegisterGlobalFunction("removeEventListener", &DummyMethod);
154 
155 #if !defined(NDEBUG) && defined(USING_JSC)
156  RegisterGlobalFunction("gc", &GC);
157 #endif
158 
159  LocalVar<JsValue> document(
160  impl_->document.WrapInstance(js::dom::Document::CreateGlobalDocument()));
161  SetMemberRaw(JsEngine::Instance()->global_handle(), "document", document);
162 
163  CreateInstance("console", &impl_->console);
164  CreateInstance("location", &impl_->location);
165  CreateInstance("navigator", &impl_->navigator);
166 
167  js::Base64::Install();
168  js::Timeouts::Install();
169 
170  // Run the script directly since we are initializing, so this is
171  // effectively the event thread.
172  JsManagerImpl* manager = JsManagerImpl::Instance();
173  CHECK(RunScript(manager->GetPathForStaticFile("shaka-player.compiled.js")));
174 }
