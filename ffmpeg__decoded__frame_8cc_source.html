52 size_t FFmpegDecodedFrame::EstimateSize() const {
53  size_t size = sizeof(*this) + sizeof(*frame_);
54  for (int i = AV_NUM_DATA_POINTERS; i; i--) {
55  if (frame_->buf[i - 1])
56  size += frame_->buf[i - 1]->size;
57  }
58  for (int i = frame_->nb_extended_buf; i; i--)
59  size += frame_->extended_buf[i - 1]->size;
60  for (int i = frame_->nb_side_data; i; i--)
61  size += frame_->side_data[i - 1]->size;
62  return size;
63 }
