118 void TextTrack::RemoveCue(VTTCue* cue) {
119  // Locate cue in list.
120  auto cues = impl_->GetMemberVariable(&JSTextTrack::cues);
121  auto inner_cue_iter = std::find(cues.begin(), cues.end(), cue);
122  CHECK(inner_cue_iter != cues.end())
123  << "Can only remove cues retrieved from cues list";
124  if (inner_cue_iter != cues.end()) {
125  RefPtr<js::VTTCue> inner_cue = *inner_cue_iter;
126  impl_->CallInnerMethod(&JSTextTrack::RemoveCue, inner_cue);
127  }
128 }
