72 void LockedFrameList::WaitToDeleteFrames(
73  const std::unordered_set<const BaseFrame*>& frames) {
74  std::unique_lock<Mutex> lock(mutex_);
75 
76 #ifndef NDEBUG
77  for (auto& locked_frame : frames_) {
78  if (frames.count(locked_frame.frame) > 0)
79  CHECK_NE(locked_frame.locked_thread, std::this_thread::get_id())
80  << "Cannot delete a frame in used by this thread.";
81  }
82 #endif
83 
84  auto has_locked = [&]() {
85  for (auto& locked_frame : frames_) {
86  if (frames.count(locked_frame.frame) > 0)
87  return true;
88  }
89  return false;
90  };
91  while (has_locked()) {
92  cond_.ResetAndWaitWhileUnlocked(lock);
93  }
94 }

96 void LockedFrameList::UnguardFrame(const BaseFrame* frame) {

98  for (auto it = frames_.begin(); it != frames_.end(); it++) {
99  if (it->frame == frame) {
