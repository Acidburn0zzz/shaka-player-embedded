68 void DemuxerThread::AppendData(double timestamp_offset, double window_start,
69  double window_end, const uint8_t* data,
70  size_t data_size,
71  std::function<void(Status)> on_complete) {
72  DCHECK(data);
73  DCHECK_GT(data_size, 0u);
74 
75  std::unique_lock<Mutex> lock(mutex_);
76  DCHECK(input_.empty()); // Should not be performing an update.
77  input_.SetBuffer(data, data_size);
78  processor_->SetTimestampOffset(timestamp_offset);
79  window_start_ = window_start;
80  window_end_ = window_end;
81  cur_data_ = data;
82  cur_size_ = data_size;
83  on_complete_ = std::move(on_complete);
84 
85  new_data_.SignalAll();

88 void DemuxerThread::ThreadMain() {
89  using namespace std::placeholders; // NOLINT
90  auto on_read = std::bind(&DemuxerThread::OnRead, this, _1, _2);
91  auto on_reset_read = std::bind(&DemuxerThread::OnResetRead, this);
92  const Status init_status =
93  processor_->InitializeDemuxer(on_read, on_reset_read);
94  if (init_status != Status::Success) {
95  // If we get an error before we append the first segment, then we won't have
96  // a callback yet, so we have nowhere to send the error to. Wait until we
97  // get the first segment.
98  std::unique_lock<Mutex> lock(mutex_);
99  if (!on_complete_ && !shutdown_) {
100  new_data_.ResetAndWaitWhileUnlocked(lock);
101  }
102  CallOnComplete(init_status);
103  return;
104  }
105 
106  // The demuxer initialization will read the init segment and load any metadata
107  // it needs.
108  on_load_meta_();
109 
110  while (!shutdown_) {
111  std::unique_ptr<BaseFrame> frame;
112  const Status status = processor_->ReadDemuxedFrame(&frame);
113  if (status != Status::Success) {
114  std::unique_lock<Mutex> lock(mutex_);
115  CallOnComplete(status);
116  break;
117  }
118 
119  {
120  std::unique_lock<Mutex> lock(mutex_);
121  if (frame->pts < window_start_ ||
122  frame->pts + frame->duration > window_end_) {
123  need_key_frame_ = true;
124  VLOG(2) << "Dropping frame outside append window, pts=" << frame->pts;
125  continue;
126  }
127  if (need_key_frame_) {
128  if (frame->is_key_frame) {
129  need_key_frame_ = false;
130  } else {
131  VLOG(2) << "Dropping frame while looking for key frame, pts="
132  << frame->pts;
133  continue;
134  }
135  }
136  }
137  stream_->GetDemuxedFrames()->AppendFrame(std::move(frame));
138  }
139 }

141 size_t DemuxerThread::OnRead(uint8_t* data, size_t data_size) {
142  std::unique_lock<Mutex> lock(mutex_);
143 
144  // If we get called with no input, then we are done processing the previous
145  // data, so call on_complete. This is a no-op if there is no callback.
146  if (input_.empty()) {
147  CallOnComplete(Status::Success);
148  if (!shutdown_) {
149  new_data_.ResetAndWaitWhileUnlocked(lock);
150  }
151  }
152 
153  if (shutdown_)
154  return 0;
155 
156  const size_t bytes_read = input_.Read(data, data_size);
157  VLOG(3) << "ReadCallback: Read " << bytes_read << " bytes from stream.";
158  return bytes_read;
159 }

166 void DemuxerThread::CallOnComplete(Status status) {
167  if (on_complete_) {
168  // on_complete must be invoked on the event thread.
169  JsManagerImpl::Instance()->MainThread()->AddInternalTask(
170  TaskPriority::Internal, "Append done",
171  PlainCallbackTask(std::bind(on_complete_, status)));
172  on_complete_ = std::function<void(Status)>();
173  }
