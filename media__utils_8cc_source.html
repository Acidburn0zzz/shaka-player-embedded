68 bool ParseMimeType(const std::string& source, std::string* type,
69  std::string* subtype,
70  std::unordered_map<std::string, std::string>* params) {
71  // Extract type.
72  const auto type_end = source.find('/');
73  if (type_end == std::string::npos)
74  return false;
75  *type = util::TrimAsciiWhitespace(substr_end(source, 0, type_end));
76  if (!IsToken(*type))
77  return false;
78 
79  // Extract subtype.
80  const auto subtype_end = source.find(';', type_end);
81  *subtype =
82  util::TrimAsciiWhitespace(substr_end(source, type_end + 1, subtype_end));
83  if (!IsToken(*subtype))
84  return false;
85 
86  // Extract parameters.
87  params->clear();
88  auto param_end = subtype_end;
89  while (param_end != std::string::npos) {
90  const auto name_end = source.find('=', param_end);
91  if (name_end == std::string::npos)
92  return false;
93 
94  const std::string param_name =
95  util::TrimAsciiWhitespace(substr_end(source, param_end + 1, name_end));
96  if (!IsToken(param_name))
97  return false;
98 
99  const auto value_start =
100  source.find_first_not_of(kWhitespaceCharacters, name_end + 1);
101  std::string value;
102  if (value_start == std::string::npos) {
103  param_end = std::string::npos;
104  value = "";
105  } else if (source[value_start] == '"') {
106  const auto str_end = source.find('"', value_start + 1);
107  if (str_end == std::string::npos)
108  return false;
109  value = substr_end(source, value_start + 1, str_end);
110  param_end = source.find(';', str_end);
111 
112  const std::string extra =
113  util::TrimAsciiWhitespace(substr_end(source, str_end + 1, param_end));
114  if (!extra.empty())
115  return false;
116  } else {
117  param_end = source.find(';', name_end);
118  value = util::TrimAsciiWhitespace(
119  substr_end(source, name_end + 1, param_end));
120  if (!IsToken(value))
121  return false;
122  }
123  params->emplace(util::ToAsciiLower(param_name), value);
124  }
125 
126  return true;
127 }

129 bool IsTypeSupported(const std::string& container, const std::string& codecs,
130  int width, int height) { // NOLINT(misc-unused-parameters)
131  if (codecs.find(',') != std::string::npos) {
132  VLOG(1) << "Multiplexed streams not supported.";
133  return false;
134  }
135 
136  const std::string norm_container = NormalizeContainer(container);
137  if (!av_find_input_format(norm_container.c_str())) {
138  VLOG(1) << "Container '" << norm_container << "' (normalized from '"
139  << container << "') is not supported";
140  return false;
141  }
142 
143  const std::string norm_codec = NormalizeCodec(codecs);
144  if (!codecs.empty() && !avcodec_find_decoder_by_name(norm_codec.c_str())) {
145  VLOG(1) << "Codec '" << norm_codec << "' (normalized from '" << codecs
146  << "') is not supported";
147  return false;
148  }
149 
150 #ifdef FORCE_HARDWARE_DECODE
151  if (!DoesHardwareSupportCodec(codecs, width, height)) {
152  VLOG(1) << "Codec '" << codecs << "' isn't supported by the hardware.";
153  return false;
154  }
155 #endif
156 
157  return true;
158 }

160 bool ParseMimeAndCheckSupported(const std::string& mime_type,
161  SourceType* source_type, std::string* container,
162  std::string* codec) {
163  std::string type;
164  std::unordered_map<std::string, std::string> params;
165  if (!ParseMimeType(mime_type, &type, container, &params))
166  return false;
167  const long width = atol(params["width"].c_str()); // NOLINT
168  const long height = atol(params["height"].c_str()); // NOLINT
169  if (!IsTypeSupported(*container, params["codecs"], width, height))
170  return false;
171 
172  if (type == "video") {
173  *source_type = SourceType::Video;
174  } else if (type == "audio") {
175  *source_type = SourceType::Audio;
176  } else {
177  VLOG(1) << "Non-audio/video MIME given '" << mime_type << "'";
178  return false;
179  }
180 
181  *codec = params["codecs"];
182  return true;
183 }

185 std::string NormalizeContainer(const std::string& container) {
186  for (const auto& entry : kContainerMap)
187  if (container == entry.source)
188  return entry.dest;
189  return container;

192 std::string NormalizeCodec(const std::string& codec) {
193  const std::string simple_codec = codec.substr(0, codec.find('.'));
194  for (const auto& entry : kCodecMap)
195  if (simple_codec == entry.source)
196  return entry.dest;
197  return simple_codec;

201 BufferedRanges IntersectionOfBufferedRanges(
202  const std::vector<BufferedRanges>& sources) {
203  if (sources.empty())
204  return BufferedRanges();
205 
206  BufferedRanges accumulated = sources[0];
207  for (auto& source : sources) {
208  BufferedRanges temp;
209  size_t acc_i = 0, source_i = 0;
210 
211  while (acc_i < accumulated.size() && source_i < source.size()) {
212  const double start =
213  std::max(accumulated[acc_i].start, source[source_i].start);
214  const double end = std::min(accumulated[acc_i].end, source[source_i].end);
215  if (end > start)
216  temp.emplace_back(start, end);
217 
218  if (accumulated[acc_i].end < source[source_i].end)
219  acc_i++;
220  else
221  source_i++;
222  }
223 
224  accumulated.swap(temp);
225  }
226 
227  return accumulated;
228 }
