84 bool RunScriptImpl(const std::string& path, Handle<JsString> source) {
85  v8::Isolate* isolate = v8::Isolate::GetCurrent();
86  auto context = isolate->GetCurrentContext();
87  v8::HandleScope handle_scope(GetIsolate());
88  v8::ScriptOrigin origin(ToJsValue(path));
89 
90  // Compile the script.
91  v8::TryCatch trycatch(isolate);
92  v8::MaybeLocal<v8::Script> maybe_script =
93  v8::Script::Compile(context, source, &origin);
94  v8::Local<v8::Script> script;
95  if (!maybe_script.ToLocal(&script) || script.IsEmpty()) {
96  LOG(ERROR) << "Error loading script " << path;
97  return false;
98  }
99 
100  // Run the script. Run() returns the return value from the script, which
101  // will be empty if the script failed to execute.
102  if (script->Run(context).IsEmpty()) {
103  OnUncaughtException(trycatch.Exception(), false);
104  return false;
105  }
106  return true;
107 }

111 std::vector<std::string> GetMemberNames(Handle<JsObject> object) {
112  std::vector<std::string> names;
113 
114  v8::Isolate* isolate = GetIsolate();
115  v8::Local<v8::Context> context = isolate->GetCurrentContext();
116 
117  v8::MaybeLocal<v8::Array> maybe_field = object->GetOwnPropertyNames(context);
118  v8::Local<v8::Array> array;
119  if (maybe_field.ToLocal(&array)) {
120  for (size_t i = 0; i < array->Length(); i++) {
121  v8::Local<v8::Value> name = GetArrayIndexRaw(array, i);
122  names.push_back(ConvertToString(name));
123  }
124  }
125 
126  return names;
127 }

156 bool InvokeConstructor(Handle<JsFunction> ctor, int argc,
157  LocalVar<JsValue>* argv,
158  LocalVar<JsValue>* result_or_except) {
159  v8::Isolate* isolate = GetIsolate();
160  v8::Local<v8::Context> context = isolate->GetCurrentContext();
161  v8::EscapableHandleScope handles(isolate);
162 
163  v8::TryCatch trycatch(isolate);
164  v8::MaybeLocal<v8::Object> result = ctor->NewInstance(context, argc, argv);
165  if (result.IsEmpty()) {
166  *result_or_except = handles.Escape(trycatch.Exception());
167  return false;
168  }
169  *result_or_except = handles.Escape(result.ToLocalChecked());
170  return true;
171 }

173 bool InvokeMethod(Handle<JsFunction> method, Handle<JsObject> that, int argc,
174  LocalVar<JsValue>* argv,
175  LocalVar<JsValue>* result_or_except) {
176  v8::Isolate* isolate = GetIsolate();
177  v8::Local<v8::Context> context = isolate->GetCurrentContext();
178  v8::EscapableHandleScope handles(isolate);
179 
180  if (that.IsEmpty())
181  that = context->Global();
182 
183  v8::TryCatch trycatch(isolate);
184  v8::MaybeLocal<v8::Value> result = method->Call(context, that, argc, argv);
185  if (result.IsEmpty()) {
186  *result_or_except = handles.Escape(trycatch.Exception());
187  return false;
188  }
189  *result_or_except = handles.Escape(result.ToLocalChecked());
190  return true;
191 }

313 JSValueType GetValueType(Handle<JsValue> value) {
314  if (value.IsEmpty())

318  if (value->IsNull())

320  if (value->IsBoolean())
321  return JSValueType::Boolean;
322  if (value->IsNumber())
323  return JSValueType::Number;
324  if (value->IsString())
325  return JSValueType::String;
326  if (value->IsSymbol())
327  return JSValueType::Symbol;
328  if (value->IsFunction())
329  return JSValueType::Function;
330  if (value->IsArray())
331  return JSValueType::Array;
332  if (value->IsPromise())
333  return JSValueType::Promise;
334  if (value->IsBooleanObject())
335  return JSValueType::BooleanObject;
336  if (value->IsNumberObject())
337  return JSValueType::NumberObject;
338  if (value->IsStringObject())
339  return JSValueType::StringObject;
340  if (value->IsArrayBuffer())
341  return JSValueType::ArrayBuffer;
342  if (value->IsInt8Array())
343  return JSValueType::Int8Array;
344  if (value->IsUint8Array())
345  return JSValueType::Uint8Array;
346  if (value->IsUint8ClampedArray())
347  return JSValueType::Uint8ClampedArray;
348  if (value->IsInt16Array())
349  return JSValueType::Int16Array;
350  if (value->IsUint16Array())
351  return JSValueType::Uint16Array;
352  if (value->IsInt32Array())
353  return JSValueType::Int32Array;
354  if (value->IsUint32Array())
355  return JSValueType::Uint32Array;
356  if (value->IsFloat32Array())
357  return JSValueType::Float32Array;
358  if (value->IsFloat64Array())
359  return JSValueType::Float64Array;
360  if (value->IsDataView())
361  return JSValueType::DataView;
362  if (value->IsObject())
363  return JSValueType::OtherObject;
364 
365  // The only thing a value can be is a primitive or an object. We should
366  // be checking every primitive above so it should be an object.
367  LOG(WARNING) << "Unknown JavaScript value given=" << ConvertToString(value);
