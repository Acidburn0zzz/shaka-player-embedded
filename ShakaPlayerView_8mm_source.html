55  }

67  bufferingCallbackLocal(is_buffering);

69  }

168 - (CGImageRef)drawFrame {
169  double delay = ShakaRenderLoopDelay;
170  auto frame = _video->DrawFrame(&delay);
171  if (!frame.valid())
172  return nullptr;
173  if (frame.pixel_format() == shaka::PixelFormat::VIDEO_TOOLBOX) {
174  CGImage *ret = nullptr;
175  uint8_t *data = const_cast<uint8_t *>(frame.data()[3]);
176  const long status =
177  VTCreateCGImageFromCVPixelBuffer(reinterpret_cast<CVPixelBufferRef>(data), nullptr, &ret);
178  if (status != 0) {
179  // Create a shaka::Error object, so it can be dispatched via the Client.
180  std::string message = "VTCreateCGImageFromCVPixelBuffer error status ";
181  message += std::to_string(status);
182  shaka::Error err = shaka::Error(shaka::ErrorType::NonShakaError, message);
183  _client.OnError(err, @"Error transferring frame to CGImage");
184  return nullptr;
185  }
186  return ret;
187  }
188 
189  if (!frame.ConvertTo(shaka::PixelFormat::RGB24))
190  return nullptr;
191 
192  // Get size.
193  const int align = 16;
194  CFIndex size = av_image_get_buffer_size(AV_PIX_FMT_RGB24, frame.width(), frame.height(), align);
195 
196  // TODO: Handle padding.
197 
198  // Create a CFData object to wrap the raw frame data. This is a no-copy
199  // CFData, since this instance does not own the frame data.
200  CFDataRef data =
201  CFDataCreateWithBytesNoCopy(kCFAllocatorDefault, frame.data()[0], size, kCFAllocatorNull);
202 
203  // Make a CGDataProvider object to distribute the data to the CGImage.
204  // This copies the data into the CGImage, so the CGImage becomes
205  // responsible for managing its lifecycle.
206  CGDataProviderRef provider = CGDataProviderCreateWithCFData(data);
207 
208  // CGColorSpaceCreateDeviceRGB makes a device-specific colorSpace, so use a
209  // standardized one instead.
210  CGColorSpaceRef color_space = CGColorSpaceCreateWithName(kCGColorSpaceSRGB);
211 
212  // Create a CGImage.
213  const size_t bits_per_pixel = 24;
214  const size_t bits_per_component = 8;
215  const size_t bytes_per_row = frame.linesize()[0];
216  const bool should_interpolate = false;
217  CGImage *image = CGImageCreate(frame.width(), frame.height(), bits_per_component, bits_per_pixel,
218  bytes_per_row, color_space, kCGBitmapByteOrderDefault, provider,
219  nullptr, should_interpolate, kCGRenderingIntentDefault);
220 
221  // Dispose of temporary data.
222  CGColorSpaceRelease(color_space);
223  CGDataProviderRelease(provider);
224 
225  return image;
226 }

228 - (void)renderLoop:(NSTimer *)timer {
229  @synchronized(self) {
230  if (CGImageRef image = [self drawFrame]) {
231  _imageLayer.contents = (__bridge_transfer id)image;
232 
233  // Fit image in frame.
234  int imageWidth = CGImageGetWidth(image);
235  int imageHeight = CGImageGetHeight(image);
236  int frameWidth = static_cast<int>(self.bounds.size.width);
237  int frameHeight = static_cast<int>(self.bounds.size.height);
238  shaka::ShakaRect rect =
239  shaka::FitVideoToWindow(imageWidth, imageHeight, frameWidth, frameHeight, 0, 0);
240  _imageLayer.frame = CGRectMake(rect.x, rect.y, rect.w, rect.h);
241  }
242 
243  if (self.closedCaptions) {
244  BOOL sizeChanged = _textLayer.frame.size.width != _imageLayer.frame.size.width ||
245  _textLayer.frame.size.height != _imageLayer.frame.size.height;
246 
247  _textLayer.hidden = NO;
248  _textLayer.frame = _imageLayer.frame;
249 
250  if (_remakeTextLayer || sizeChanged) {
251  // Remake text cues, since either they changed or the size of the bounds changed.
252  [self remakeTextCues];
253  _remakeTextLayer = NO;
254  }
255  } else {
256  _textLayer.hidden = YES;
257  }
258  }
259 }

261 - (void)remakeTextCues {
262  auto text_tracks = _video->TextTracks();
263  auto cues = text_tracks[0].cues();
264 
265  // TODO: Once activeCues is implemented, use that instead.
266  // TODO: This could be more efficient if I take advantage of how the cues are
267  // guaranteed to be sorted, so I can safely stop once the cues overtake the
268  // time.
269  double time = self.currentTime;
270  auto callback = [time](shaka::VTTCue *cue) {
271  return cue->startTime() <= time && cue->endTime() > time;
272  };
273  auto begin = cues.begin();
274  auto end = cues.end();
275  std::vector<shaka::VTTCue *> activeCues;
276  std::copy_if(begin, end, std::back_inserter(activeCues), callback);
277 
278  // TODO: Try recycling old text layers which contain the correct text contents.
279  // This will make the layers do a move animation instead of a fade animation when appropriate,
280  // and also might be more performant.
281  for (CALayer *layer in [_textLayer.sublayers copy])
282  [layer removeFromSuperlayer];
283 
284  // Use the system font for body. The ensures that if the user changes their font size, it will use
285  // that font size.
286  UIFont *font = [UIFont preferredFontForTextStyle:UIFontTextStyleBody];
287 
288  // Create CATextLayers for cues.
289  NSMutableArray<CATextLayer *> *cueLayers = [NSMutableArray array];
290  for (unsigned long i = 0; i < activeCues.size(); i++) {
291  // Read the cues in inverse order, so the oldest cue is at the bottom.
292  shaka::VTTCue *cue = activeCues[activeCues.size() - i - 1];
293  NSString *cueText = [NSString stringWithUTF8String:cue->text().c_str()];
294 
295  for (NSString *line in [cueText componentsSeparatedByString:@"\n"]) {
296  CATextLayer *cueLayer = [[CATextLayer alloc] init];
297  cueLayer.string = line;
298  cueLayer.font = CGFontCreateWithFontName(static_cast<CFStringRef>(font.fontName));
299  cueLayer.fontSize = font.pointSize;
300  cueLayer.backgroundColor = [UIColor blackColor].CGColor;
301  cueLayer.foregroundColor = [UIColor whiteColor].CGColor;
302  cueLayer.alignmentMode = kCAAlignmentCenter;
303 
304  // TODO: Take into account direction setting, snap to lines, position align, etc.
305 
306  // Determine size of cue line.
307  CGFloat width = static_cast<CGFloat>(cue->size() * (_textLayer.bounds.size.width) * 0.01);
308  NSStringDrawingOptions options = NSStringDrawingUsesLineFragmentOrigin;
309  // TODO: Sometimes, if the system font size is set high, this will make very wrong estimates.
310  CGSize effectiveSize = [line boundingRectWithSize:CGSizeMake(width, 9999)
311  options:options
312  attributes:@{NSFontAttributeName: font}
313  context:nil]
314  .size;
315  // The size estimates don't always seem to be 100% accurate, so this adds a bit to the width.
316  width = ceil(effectiveSize.width) + 16;
317  CGFloat height = ceil(effectiveSize.height);
318  CGFloat x = (_textLayer.bounds.size.width - width) * 0.5f;
319  cueLayer.frame = CGRectMake(x, 0, width, height);
320 
321  [cueLayers addObject:cueLayer];
322  [_textLayer addSublayer:cueLayer];
323  }
324  }
325 
326  // Determine y positions for the CATextLayers.
327  CGFloat y = _textLayer.bounds.size.height;
328  for (unsigned long i = 0; i < cueLayers.count; i++) {
329  // Read the cue layers in inverse order, since they are being drawn from the bottom up.
330  CATextLayer *cueLayer = cueLayers[cueLayers.count - i - 1];
331 
332  CGSize size = cueLayer.frame.size;
333  y -= size.height;
334  cueLayer.frame = CGRectMake(cueLayer.frame.origin.x, y, size.width, size.height);
335  }
336 }

551 - (void)postLoadOperations {
552  _imageLayer.contents = nil;
553 
554  // Start up the render loop.
555  SEL renderLoopSelector = @selector(renderLoop:);
556  _renderLoopTimer = [NSTimer scheduledTimerWithTimeInterval:ShakaRenderLoopDelay
557  target:self
558  selector:renderLoopSelector
559  userInfo:nullptr
560  repeats:YES];
561 
562  auto textTracks = _video->TextTracks();
563  if (!textTracks.empty()) {
564  _remakeTextLayer = YES;
565 
566  // Make a weak reference to self, to avoid a retain cycle.
567  __weak ShakaPlayerView *weakSelf = self;
568 
569  // TODO: maybe this should be handled in video or player?
570  auto onCueChange = [weakSelf]() {
571  // To avoid changing layers on a background thread, do the actual cue
572  // operations in the render loop.
573  __strong ShakaPlayerView *strongSelf = weakSelf;
574  strongSelf->_remakeTextLayer = YES;
575  };
576  textTracks[0].SetCueChangeEventListener(onCueChange);
577  }
578 }

580 - (void)unload {
581  // Stop the render loop before acquiring the mutex, to avoid deadlocks.
582  [_renderLoopTimer invalidate];
583 
584  @synchronized(self) {
585  _imageLayer.contents = nil;
586  for (CALayer *layer in [_textLayer.sublayers copy])
587  [layer removeFromSuperlayer];
588  const auto unloadResults = _player->Unload();
589  if (unloadResults.has_error()) {
590  _client.OnError(unloadResults.error(), @"Error unloading asset");
591  }
592  }
593 }
