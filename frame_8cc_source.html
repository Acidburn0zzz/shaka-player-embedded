100 bool Frame::ConvertTo(PixelFormat format) {
101  if (!impl_)
102  return false;
103  if (impl_->format == format)
104  return true;
105 
106  AVPixelFormat pix_fmt;
107  switch (format) {
108  case PixelFormat::YUV420P:
109  pix_fmt = AV_PIX_FMT_YUV420P;
110  break;
111  case PixelFormat::NV12:
112  pix_fmt = AV_PIX_FMT_NV12;
113  break;
114  case PixelFormat::RGB24:
115  pix_fmt = AV_PIX_FMT_RGB24;
116  break;
117 
118  case PixelFormat::VIDEO_TOOLBOX:
119  LOG(ERROR) << "Cannot convert to a hardware-accelerated format.";
120  return false;
121 
122  default:
123  LOG(FATAL) << "Unknown pixel format: "
124  << static_cast<std::underlying_type<PixelFormat>::type>(
125  format);
126  }
127 
128  if (!impl_->converter.ConvertFrame(impl_->frame, &impl_->frame_data,
129  &impl_->frame_linesize, pix_fmt)) {
130  return false;
131  }
132 
133  impl_->format = format;
134  return true;
135 }
