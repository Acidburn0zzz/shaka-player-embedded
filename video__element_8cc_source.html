56 void HTMLVideoElement::ThreadMain() {
57  double lastTime = CurrentTime();
58  while (!shutdown_.load(std::memory_order_acquire)) {
59  const double time = CurrentTime();
60 
61  if (pipeline_status_ == media::PipelineStatus::Playing) {
62  CheckForCueChange(time, lastTime);
63  lastTime = time;
64  }
65 
66  clock_->SleepSeconds(0.25);
67  }
68 }

70 void HTMLVideoElement::OnReadyStateChanged(
71  media::MediaReadyState new_ready_state) {
72  DCHECK(media_source_ ? new_ready_state != media::HAVE_NOTHING
73  : new_ready_state == media::HAVE_NOTHING);
74  if (ready_state == new_ready_state)
75  return;
76 
77  if (ready_state < media::HAVE_METADATA &&
78  new_ready_state >= media::HAVE_METADATA) {
79  ScheduleEvent<events::Event>(EventType::LoadedMetaData);
80  }
81  if (ready_state < media::HAVE_CURRENT_DATA &&
82  new_ready_state >= media::HAVE_CURRENT_DATA) {
83  ScheduleEvent<events::Event>(EventType::LoadedData);
84  }
85  if (ready_state < media::HAVE_ENOUGH_DATA &&
86  new_ready_state >= media::HAVE_ENOUGH_DATA) {
87  ScheduleEvent<events::Event>(EventType::CanPlay);
88  }
89  if (ready_state >= media::HAVE_FUTURE_DATA &&
90  new_ready_state < media::HAVE_FUTURE_DATA &&
91  new_ready_state != media::HAVE_NOTHING) {
92  ScheduleEvent<events::Event>(EventType::Waiting);
93  }
94 
95  ScheduleEvent<events::Event>(EventType::ReadyStateChange);
96  ready_state = new_ready_state;
97 }

99 void HTMLVideoElement::OnPipelineStatusChanged(media::PipelineStatus status) {
100  if (status == pipeline_status_) {
101  // If we get another seeking status change, we still fire the 'seeking'
102  // event since the current time changed.
103  if (status == media::PipelineStatus::SeekingPlay ||
104  status == media::PipelineStatus::SeekingPause) {
105  ScheduleEvent<events::Event>(EventType::Seeking);
106  }
107  return;
108  }
109 
110  switch (status) {
111  case media::PipelineStatus::Initializing:
112  ScheduleEvent<events::Event>(EventType::Emptied);
113  break;
114  case media::PipelineStatus::Playing:
115  if (pipeline_status_ == media::PipelineStatus::Paused) {
116  ScheduleEvent<events::Event>(EventType::Play);
117  } else if (pipeline_status_ == media::PipelineStatus::SeekingPlay) {
118  ScheduleEvent<events::Event>(EventType::Seeked);
119  } else {
120  DCHECK(pipeline_status_ == media::PipelineStatus::Stalled ||
121  pipeline_status_ == media::PipelineStatus::Initializing);
122  }
123  ScheduleEvent<events::Event>(EventType::Playing);
124  break;
125  case media::PipelineStatus::Paused:
126  if (pipeline_status_ == media::PipelineStatus::Playing ||
127  pipeline_status_ == media::PipelineStatus::Stalled) {
128  ScheduleEvent<events::Event>(EventType::Pause);
129  } else if (pipeline_status_ == media::PipelineStatus::SeekingPause) {
130  ScheduleEvent<events::Event>(EventType::Seeked);
131  } else {
132  DCHECK_EQ(pipeline_status_, media::PipelineStatus::Initializing);
133  }
134  break;
135  case media::PipelineStatus::Stalled:
136  if (pipeline_status_ == media::PipelineStatus::Playing)
137  ScheduleEvent<events::Event>(EventType::Pause);
138  break;
139  case media::PipelineStatus::SeekingPlay:
140  if (pipeline_status_ == media::PipelineStatus::Playing)
141  ScheduleEvent<events::Event>(EventType::Pause);
142  FALL_THROUGH_INTENDED;
143  case media::PipelineStatus::SeekingPause:
144  ScheduleEvent<events::Event>(EventType::Seeking);
145  break;
146  case media::PipelineStatus::Ended:
147  if (pipeline_status_ == media::PipelineStatus::Playing) {
148  ScheduleEvent<events::Event>(EventType::Pause);
149  } else if (pipeline_status_ == media::PipelineStatus::SeekingPlay ||
150  pipeline_status_ == media::PipelineStatus::SeekingPause) {
151  ScheduleEvent<events::Event>(EventType::Seeked);
152  }
153  ScheduleEvent<events::Event>(EventType::Ended);
154  break;
155  case media::PipelineStatus::Errored:
156  ScheduleEvent<events::Event>(EventType::Error);
157  if (!error)
158  error = new MediaError(MEDIA_ERR_DECODE, "Unknown media error");
159  break;
160  }
161 
162  pipeline_status_ = status;
163 }

182 Promise HTMLVideoElement::SetMediaKeys(RefPtr<eme::MediaKeys> media_keys) {
183  if (!media_keys && !media_source_)
184  return Promise::Resolved();
185  if (!media_source_) {
186  return Promise::Rejected(JsError::DOMException(
187  InvalidStateError, "Cannot set MediaKeys until after setting source"));

190  eme::Implementation* cdm = media_keys ? media_keys->GetCdm() : nullptr;
191  media_source_->GetController()->SetCdm(cdm);
192  this->media_keys = media_keys;
193  return Promise::Resolved();

231 ExceptionOr<void> HTMLVideoElement::SetSource(const std::string& src) {
232  // Unload any previous MediaSource objects.
233  Load();
234 
235  DCHECK(!media_source_);
236  if (src.empty())
237  return {};
238 
239  media_source_ = MediaSource::FindMediaSource(src);
240  if (media_source_) {
241  media_source_->OpenMediaSource(this);
242  media_source_->GetController()->SetVolume(is_muted_ ? 0 : volume_);
243  if (autoplay || will_play_)
244  media_source_->GetController()->GetPipelineManager()->Play();
245  } else {
246  // We don't support direct URL playback, only MediaSource playback.
247  return JsError::DOMException(NotSupportedError,
248  "Unknown MediaSource URL given.");
249  }
250  return {};
251 }
