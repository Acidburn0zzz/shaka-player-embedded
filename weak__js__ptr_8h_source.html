52  WeakJsPtr(const WeakJsPtr& other) {

58  WeakJsPtr(WeakJsPtr&& other) {
59  // There are problems using the v8::Global<T> move constructor. So just

161  template <typename U>
162  void ResetInternal(const U& other) {
163 #if defined(USING_V8)
164  // Don't do anything if they are both empty. This allows using WeakJsPtr
165  // in a background thread where GetIsolate() would fail a CHECK.
166  if (ptr_.IsEmpty() && other.IsEmpty())
167  return;
168 
169  // Mark the Global as weak so the object can be destroyed even though we
170  // hold a pointer. If |this| is alive, the HeapTracer will call MarkAlive.
171  ptr_.Reset(GetIsolate(), other);
172  if (!ptr_.IsEmpty())
173  ptr_.SetWeak();
174 #else
175  ptr_ = other;
176 #endif
177  }
