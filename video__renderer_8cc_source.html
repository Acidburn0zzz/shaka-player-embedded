46 Frame VideoRenderer::DrawFrame(int* dropped_frame_count, bool* is_new_frame,
47  double* delay) {
48  std::unique_lock<Mutex> lock(mutex_);
49 
50  // Discard any old frames, except when seeking.
51  if (!is_seeking_ && prev_time_ >= 0)
52  stream_->GetDecodedFrames()->Remove(0, prev_time_ - 0.2);
53 
54  // TODO: Could this usage cause a deadlock? If a remove() is started after
55  // ideal_frame is locked, the remove() will block and getting next_frame
56  // will wait for remove().
57  const double time = get_time_();
58  // If we are seeking, use the previous frame so we display the same frame
59  // while the seek is happening.
60  auto ideal_frame = is_seeking_ && prev_time_ >= 0
61  ? stream_->GetDecodedFrames()->GetFrameNear(prev_time_)
62  : stream_->GetDecodedFrames()->GetFrameNear(time);
63  if (!ideal_frame)
64  return Frame();
65 
66  // TODO: Consider changing effective playback rate to speed up video when
67  // behind. This makes playback smoother at the cost of being more complicated
68  // and sacrificing AV sync.
69 
70  auto next_frame =
71  stream_->GetDecodedFrames()->GetFrameAfter(ideal_frame->pts);
72  const double total_delay = next_frame ? next_frame->pts - time : 0;
73  *delay = std::max(std::min(total_delay, kMaxDelay), kMinDelay);
74 
75  *is_new_frame = prev_time_ != ideal_frame->pts;
76  if (!is_seeking_) {
77  if (prev_time_ >= 0) {
78  *dropped_frame_count = stream_->GetDecodedFrames()->FramesBetween(
79  prev_time_, ideal_frame->pts);
80  }
81  prev_time_ = ideal_frame->pts;
82  }
83  return drawer_->DrawFrame(ideal_frame.get());
84 }

91 void VideoRenderer::OnSeekDone() {
92  std::unique_lock<Mutex> lock(mutex_);
93  is_seeking_ = false;
94  prev_time_ = -1;
95 
96  // Now that the seek is done, discard frames from the previous time while
97  // keeping the newly decoded frames. Don't discard too close to current time
98  // since we might discard frames that were just decoded.
99  const double time = get_time_();
100  stream_->GetDecodedFrames()->Remove(0, time - 1);
101  stream_->GetDecodedFrames()->Remove(time + 1, HUGE_VAL);
102 }
