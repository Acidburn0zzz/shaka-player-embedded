40 void IncrementIv(uint32_t count, std::vector<uint8_t>* iv) {
41  DCHECK_EQ(iv->size(), 16u);
42  auto* counter = reinterpret_cast<uint32_t*>(&iv->at(8));
43  // The IV is a big-endian integer. ntohl will convert a big-endian number to
44  // a host byte order; htonl does the opposite. Since these are unsigned,
45  // overflow is well-defined.
46  constexpr const uint32_t max = std::numeric_limits<uint32_t>::max();
47  if (max - count < ntohl(counter[1])) {
48  counter[0] = htonl(ntohl(counter[0]) + 1);
49  }
50  counter[1] = htonl(ntohl(counter[1]) + count);
51 }

90 Status FFmpegEncodedFrame::Decrypt(eme::Implementation* cdm,
91  AVPacket* dest_packet) const {
92  DCHECK(cdm) << "Must pass a CDM";
93  DCHECK(dest_packet) << "Must pass a valid packet";
94  DCHECK_LE(packet_.size, dest_packet->size);
95  DCHECK(is_encrypted()) << "This frame isn't encrypted";
96 
97  int side_data_size;
98  uint8_t* side_data = av_packet_get_side_data(
99  &packet_, AV_PKT_DATA_ENCRYPTION_INFO, &side_data_size);
100  if (!side_data) {
101  LOG(ERROR) << "Unable to get side data from packet.";
102  return Status::UnknownError;
103  }
104 
105  std::unique_ptr<AVEncryptionInfo, void (*)(AVEncryptionInfo*)> enc_info(
106  av_encryption_info_get_side_data(side_data, side_data_size),
107  &av_encryption_info_free);
108  if (!enc_info) {
109  LOG(ERROR) << "Could not allocate new encryption info structure.";
110  return Status::OutOfMemory;
111  }
112 
113  eme::EncryptionScheme scheme;
114  switch (enc_info->scheme) {
115  case kCencScheme:
116  if (enc_info->crypt_byte_block != 0 || enc_info->skip_byte_block != 0) {
117  LOG(ERROR) << "Cannot specify encryption pattern with 'cenc' scheme.";
118  return Status::InvalidContainerData;
119  }
120  scheme = eme::EncryptionScheme::AesCtr;
121  break;
122  case kCensScheme:
123  scheme = eme::EncryptionScheme::AesCtr;
124  break;
125  case kCbc1Scheme:
126  if (enc_info->crypt_byte_block != 0 || enc_info->skip_byte_block != 0) {
127  LOG(ERROR) << "Cannot specify encryption pattern with 'cbc1' scheme.";
128  return Status::InvalidContainerData;
129  }
130  scheme = eme::EncryptionScheme::AesCbc;
131  break;
132  case kCbcsScheme:
133  scheme = eme::EncryptionScheme::AesCbc;
134  break;
135 
136  default:
137  LOG(ERROR) << "Scheme 0x" << std::hex << enc_info->scheme
138  << " is unsupported";
139  return Status::NotSupported;
140  }
141 
142  if (enc_info->subsample_count == 0) {
143  const eme::DecryptStatus decrypt_status = cdm->Decrypt(
144  scheme,
145  eme::EncryptionPattern{enc_info->crypt_byte_block,
146  enc_info->skip_byte_block},
147  0, enc_info->key_id, enc_info->key_id_size, enc_info->iv,
148  enc_info->iv_size, packet_.data, packet_.size, dest_packet->data);
149  switch (decrypt_status) {
150  case eme::DecryptStatus::Success:
151  break;
152  case eme::DecryptStatus::NotSupported:
153  return Status::NotSupported;
154  case eme::DecryptStatus::KeyNotFound:
155  return Status::KeyNotFound;
156  default:
157  return Status::UnknownError;
158  }
159  } else {
160  const uint8_t* src = packet_.data;
161  uint8_t* dest = dest_packet->data;
162  size_t total_size = packet_.size;
163  size_t block_offset = 0;
164  std::vector<uint8_t> cur_iv(enc_info->iv, enc_info->iv + enc_info->iv_size);
165  for (uint32_t i = 0; i < enc_info->subsample_count; i++) {
166  const unsigned int clear_bytes =
167  enc_info->subsamples[i].bytes_of_clear_data;
168  const unsigned int protected_bytes =
169  enc_info->subsamples[i].bytes_of_protected_data;
170  if (total_size < clear_bytes ||
171  total_size - clear_bytes < protected_bytes) {
172  LOG(ERROR) << "Invalid subsample size";
173  return Status::InvalidContainerData;
174  }
175 
176  // Copy clear content first.
177  memcpy(dest, src, clear_bytes);
178  src += clear_bytes;
179  dest += clear_bytes;
180  total_size -= clear_bytes;
181 
182  // If there is nothing to decrypt, skip to the next subsample.
183  if (protected_bytes == 0)
184  continue;
185 
186  // Decrypt encrypted content.
187  const eme::DecryptStatus decrypt_status = cdm->Decrypt(
188  scheme,
189  eme::EncryptionPattern{enc_info->crypt_byte_block,
190  enc_info->skip_byte_block},
191  block_offset, enc_info->key_id, enc_info->key_id_size, cur_iv.data(),
192  cur_iv.size(), src, protected_bytes, dest);
193  switch (decrypt_status) {
194  case eme::DecryptStatus::Success:
195  break;
196  case eme::DecryptStatus::NotSupported:
197  return Status::NotSupported;
198  case eme::DecryptStatus::KeyNotFound:
199  return Status::KeyNotFound;
200  default:
201  return Status::UnknownError;
202  }
203 
204  if (enc_info->scheme == kCencScheme || enc_info->scheme == kCensScheme) {
205  uint32_t increment = 0;
206  if (enc_info->scheme == kCencScheme) {
207  // Increment the IV for each AES block we decrypted; add the
208  // block_offset to account for any partial block at the beginning.
209  increment = (block_offset + protected_bytes) / kAesBlockSize;
210  } else {
211  // Increment the IV for each encrypted block within the pattern.
212  const size_t num_blocks = protected_bytes / kAesBlockSize;
213  const size_t pattern_size =
214  enc_info->crypt_byte_block + enc_info->skip_byte_block;
215 
216  // Count the number of "pattern blocks" in the protected part of the
217  // subsample. If there is a partial pattern, include it only if it
218  // has a whole crypt_byte_block.
219  increment = (num_blocks / pattern_size) * enc_info->crypt_byte_block;
220  if (num_blocks % pattern_size >= enc_info->crypt_byte_block)
221  increment += enc_info->crypt_byte_block;
222  }
223  IncrementIv(increment, &cur_iv);
224  block_offset = (block_offset + protected_bytes) % kAesBlockSize;
225  } else if (enc_info->scheme == kCbc1Scheme) {
226  // 'cbc1' uses cipher-block-chaining, so the IV should be the last
227  // encrypted block of this subsample.
228  if (protected_bytes < kAesBlockSize ||
229  protected_bytes % kAesBlockSize != 0) {
230  LOG(ERROR) << "'cbc1' requires subsamples to be a multiple of the "
231  "AES block size.";
232  return Status::InvalidContainerData;
233  }
234  const uint8_t* block_end = src + protected_bytes;
235  cur_iv.assign(block_end - kAesBlockSize, block_end);
236  }
237  // 'cbcs' uses constant-IV.
238 
239  src += protected_bytes;
240  dest += protected_bytes;
241  total_size -= protected_bytes;
242  }
243 
244  if (total_size != 0) {
245  LOG(ERROR) << "Extra remaining data after subsample handling";
246  return Status::InvalidContainerData;
247  }
248  }
249 
250  return Status::Success;
251 }
