31 bool IsBufferedUntil(const BufferedRanges& ranges, double start_time,
32  double end_time, double duration) {
33  for (auto& range : ranges) {
34  if (range.start <= start_time + FrameBuffer::kMaxGapSize &&
35  (range.end >= end_time || end_time + kEpsilon >= duration)) {
36  return true;
37  }
38  }
39  return false;
40 }

42 bool CanPlay(const BufferedRanges& ranges, double time, double duration) {
43  return IsBufferedUntil(ranges, time, time + kNeedForPlay, duration);

72 void PipelineMonitor::ThreadMain() {
73  while (!shutdown_.load(std::memory_order_acquire)) {
74  const BufferedRanges buffered = get_buffered_();
75  const BufferedRanges decoded = get_decoded_();
76  const double time = pipeline_->GetCurrentTime();
77  const double duration = pipeline_->GetDuration();
78  const bool can_play = CanPlay(buffered, time, duration);
79  if (time >= duration) {
80  pipeline_->OnEnded();
81  } else if (can_play && IsBufferedUntil(decoded, time, time, duration)) {
82  // Don't move playhead until we have decoded at the current time. This
83  // ensures we stop for decryption errors and that we don't blindly move
84  // forward without the correct frames.
85  pipeline_->CanPlay();
86  } else {
87  pipeline_->Stalled();
88  }
89 
90  if (pipeline_->GetPipelineStatus() == PipelineStatus::Initializing) {
91  ChangeReadyState(HAVE_NOTHING);
92  } else if (can_play) {
93  ChangeReadyState(HAVE_FUTURE_DATA);
94  } else if (IsBufferedUntil(buffered, time, time, duration)) {
95  ChangeReadyState(HAVE_CURRENT_DATA);
96  } else {
97  ChangeReadyState(HAVE_METADATA);
98  }
99 
100  clock_->SleepSeconds(0.01);
101  }
102 }
